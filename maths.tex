\subsection{Introduction to groups}
\subsubsection{Magmas and properties}

\begin{definition} \label{def-magma}
    A \emph{magma} is a pair $(M, \ast)$ with $M$ a set and 
    \[
        \ast \colon M \times M \to M
    \]
    an internal composition law.
\end{definition}

\begin{definition} \label{def-magma-associative}
    A magma $(M, \ast)$ is said to be \emph{associative} if, for all $a, b, c \in M$,
    \[
        (a \ast b) \ast c = a \ast (b \ast c).
    \]
\end{definition}

\begin{definition} \label{def-magma-commutative}
    A magma $(M, \ast)$ is said to be \emph{commutative} if, for all $a, b \in M$,
    \[
        a \ast b = b \ast a.
    \]
    
\end{definition}

\begin{definition} \label{def-magma-id-element}
    A magma $(M, \ast)$ is called \emph{unital} if there exists $e \in M$ such that
    \[
        e \ast a = a = a \ast e
    \]  
    for all $a \in M$. Such an element $e$ is called an \emph{identity element}.
\end{definition}

\begin{theorem} \label{thm-magma-id-element-uniqueness}
    If $(M, \ast)$ is a unital magma, then it has a unique identity element.
\end{theorem}

\begin{proof}
    If $e, e^{\prime} \in M$ are identity elements, then $e = e \ast e^{\prime} = e^{\prime}$.
\end{proof}

\begin{notation}
    For a unital magma $(M, \ast)$, we will denote its identity element by $e_M$.
\end{notation}

\begin{definition} \label{def-magma-invertible-element}
    Let $(M, \ast)$ be a unital magma. An element $a \in M$ is said to be \emph{invertible} if there exists $b \in M$ such that $a \ast b = b \ast a = e_M$. $b$ is then called an inverse of $a$.
\end{definition}

\subsubsection{Submagmas}
\begin{definition} \label{def-magma-submagma}
    Let $(M, \ast)$ be a magma. A subset $N \subset M$ is a \emph{submagma} of $(M, \ast)$ if the operation $\ast$ is stable under $N$:
    \[
        \forall a, b \in N, \; a \ast b \in N.
    \]
\end{definition}

\begin{remark} \label{rem-magma-canonical-submagma}
    The submagma $N$ induces the canonical magma $(N, \ast_{\vert N \times N})$.
\end{remark}

\begin{remark} \label{rem-magma-submagmas-intersection}
    Any intersection of submagmas of a magma $(M, \ast)$ is a submagma of $(M, \ast)$.
\end{remark}

\subsubsection{Products of magmas}

\begin{definition} \label{def-magma-product}
    Given two magmas $(M, \ast)$ and $(N, \cdot)$, we define a canonical structure of magma on the set $M \times N$, defined by the following internal composition law:
    \[
        \big((a, b), (a^{\prime}, b^{\prime}) \big) \mapsto (a \ast a^{\prime}, b \cdot b^{\prime})
    \]
\end{definition}

\begin{proposition} \label{prop-magma-product-props}
    Let $M$ and $N$ be two magmas.
    \begin{enumerate}[label=(\roman*)]
        \item If $M$ and $N$ are associative (resp. commutative), then $M \times N$ is associative (resp. commutative),
        \item If $M$ and $N$ are unital, then $M \times N$ is unital.
    \end{enumerate}
\end{proposition}

\begin{proof}
    Omitted.
\end{proof}

\begin{proposition} \label{prop-magma-product-invertibility}
    Let $M$ and $N$ be two unital magmas. If $a \in M$ and $b \in N$ are invertible, then $(a, b) \in M \times N$ is invertible.
\end{proposition}

\begin{proof}
    If $a^{\prime} \in M$ (resp. $b^{\prime} \in N$) is an inverse of $a$ (resp. $b$), then $(a^{\prime}, b^{\prime}) \in M \times N$ is an inverse of $(a, b)$.
\end{proof}

\begin{remark}
    The previous construction can be extended to any product of magmas. The propositions \ref{prop-magma-product-props} and \ref{prop-magma-product-invertibility} then remain true.
\end{remark}

\subsubsection{Magma morphisms}

\begin{definition} \label{def-magma-morphism}
    A \emph{magma morphism} from $(M, \ast)$ to $(N, \cdot)$ is a map $f \colon M \to N$ such that:
    \[
        \forall a, b \in M, \; f(a \ast b) = f(a) \cdot f(b)
    \]
\end{definition}

\begin{proposition} \label{prop-magma-morphism-composition}
    A composition of magma morphisms is a magma morphism.
\end{proposition}

\begin{proof}
    Let $f \colon M \to M^{\prime}$ and $g \colon M^{\prime} \to M^{\prime \prime}$ be two magma morphisms. For all $a, b \in M$, we have:
    \[
        \begin{aligned}
            (g \circ f)(a \ast_{M} b)
                &= g\big( f(a \ast_{M} b) \big) \\
                &= g\big( f(a) \ast_{M^{\prime}} f(b) \big) \\
                &= g\big( f(a) \big) \ast_{M^{\prime \prime}} g\big( f(b) \big) \\
                &= (g \circ f)(a) \ast_{M^{\prime \prime}} (g \circ f)(b).
        \end{aligned}
    \]
    Therefore, $g \circ f$ is a magma morphism.
\end{proof}

\begin{theorem} \label{thm-magma-morphism-image}
    Let $f \colon (M, \ast) \to (M^{\prime}, \cdot)$ be a magma morphism, then:
    \begin{enumerate}[label=(\roman*)]
        \item If $N$ is a submagma of $M$, then $f(N)$ is a submagma of $M^{\prime}$,
        \item If $N^{\prime}$ is a submagma of $M^{\prime}$, then $f^{-1}(N^{\prime})$ is a submagma of $M$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    Firstly, for all $a, b \in N$, we have: $f(a) \cdot f(b) = f(a \ast b) \in f(N)$. Secondly, for all $a, b \in f^{-1}(N^{\prime})$, we have: $f(a \ast b) = f(a) \cdot f(b) \in N^{\prime}$.
\end{proof}

\begin{definition} \label{def-magma-isomorphism}
    A \emph{magma isomorphism} is a bijective magma morphism.
\end{definition}

\begin{theorem} \label{thm-magma-isomorphism-inverse}
    If $f$ is a magma isomorphism, then the inverse $f^{-1}$ is a magma morphism. Moreover, it is a magma isomorphism.
\end{theorem}

\begin{proof}
    Let $f \colon (M, \ast) \to (M^{\prime}, \cdot)$ be a magma isomorphism and $a, a^{\prime} \in M^{\prime}$. 
    Then
    \[
        \begin{aligned}
            f^{-1}(a \cdot a^{\prime}) 
                &= f^{-1}\Big( f\big( f^{-1}(a) \big) \cdot f\big( f^{-1}(a^{\prime}) \big) \Big) \\
                &= f^{-1}\Big( f\big( f^{-1}(a) \ast f^{-1}(a^{\prime}) \big) \Big) \\
                &= (f^{-1} \circ f) \Big( f^{-1}(a) \ast f^{-1}(a^{\prime}) \Big) \\
                &= f^{-1}(a) \ast f^{-1}(a^{\prime})
        \end{aligned}
    \]
\end{proof}

\subsubsection{Monoids and properties}

\begin{definition} \label{def-monoid}
    A \emph{monoid} is an associative unital magma.
\end{definition}

\begin{theorem} \label{thm-monoid-inverse-uniqueness}
    If $(M, \ast)$ is a monoid, then every invertible element $a \in M$ has a unique inverse, denoted by $a^{-1}$.
\end{theorem}

\begin{proof}
    Let $b, b^{\prime} \in M$ be two inverses of $a$. Then we have
    \[
        b = b \ast (a \ast b^{\prime}) = (b \ast a) \ast b^{\prime} = b^{\prime}.
    \]
\end{proof}

\begin{notation} \label{not-monoid-exponent}
    Let $(M, \ast)$ be a monoid. For all $a \in M$, we denote:
    \[
        a^0 = e_M \quad \text{and} \quad a^n = \underbrace{a \ast \dots \ast a}_{n \text{ times}} \quad (\text{for } n \geq 1).
    \]
\end{notation}

\begin{proposition}
    Given a monoid $(M, \ast)$ and $a, x, y \in M$,
    \begin{enumerate}[label=(\roman*)]
        \item If $a \ast x = a \ast y$ and $a$ is invertible, then $x = y$,
        \item If $x$ is invertible, then $x^{-1}$ is invertible and $(x^{-1})^{-1} = x$.
        \item If $x$ and $y$ are invertible, then $x \ast y$ is also invertible and $(x \ast y)^{-1} = y^{-1} \ast x^{-1}$,
        \item If $x$ is invertible, then for all $n \in \mathbb{N}, \; x^n$ is also invertible and $(x^n)^{-1} = (x^{-1})^n$. Hence we can denote $x^{-n} = (x^n)^{-1}$,
    \end{enumerate}
\end{proposition}

\begin{proof}
    \begin{enumerate}[label=(\roman*)]
        \item $x = (a^{-1} \ast a) \ast x = a^{-1} \ast (a \ast x) = a^{-1} \ast (a \ast y) = (a^{-1} \ast a) \ast y = y$.
        \item Direct consequence of the invertible element definition.
        \item $(x \ast y) \ast (y^{-1} \ast x^{-1}) = x \ast y \ast y^{-1} \ast x^{-1} = e_M$, and similarly for the converse.
        \item The result follows by combining the two preceding properties using induction.
    \end{enumerate}
\end{proof}

\subsubsection{Submonoid, product and morphism}

\begin{definition} \label{def-monoid-submonoid}
    Let $(M, \ast)$ be a monoid. A subset $N \subset M$ is a \emph{submonoid} of $(M, \ast)$ if it is a submagma of $M$ and contains the identity element.
\end{definition}

\begin{remark} \label{rem-monoid-canonical-submonoid}
    The submonoid $N$ induces the canonical monoid $(N, \ast_{\vert N \times N})$.
\end{remark}

\begin{remark} \label{rem-monoid-submonoids-intersection}
    Any intersection of submonoids of a monoid $(M, \ast)$ is a submonoid of $(M, \ast)$.
\end{remark}

\begin{proof}
    Obvious by \ref{rem-magma-submagmas-intersection} and the definition \ref{def-monoid-submonoid}
\end{proof}

\begin{proposition} \label{prop-monoid-product}
    Any product of monoids is a monoid.
\end{proposition}

\begin{proof}
    It is induced by \ref{prop-magma-product-props}.
\end{proof}

\begin{definition} \label{def-monoid-morphism}
    A \emph{monoid morphism} is a morphism between underlying magmas, such that it preserves the identity element.
\end{definition}

\begin{proposition} \label{prop-monoid-morphism-composition}
    A composition of monoid morphisms is a monoid morphism.
\end{proposition}

\begin{proof}
    By \ref{prop-magma-morphism-composition} and the fact that the invariance is stable under composition.
\end{proof}

\begin{theorem} \label{thm-monoid-morphism-image}
    Let $f \colon M \to M^{\prime}$ be a monoid morphism, then:
    \begin{enumerate}[label=(\roman*)]
        \item If $N$ is a submonoid of $M$, then $f(N)$ is a submonoid of $M^{\prime}$,
        \item If $N^{\prime}$ is a submonoid of $M^{\prime}$, then $f^{-1}(N^{\prime})$ is a submonoid of $M$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    Induced by \ref{thm-magma-morphism-image} and the fact that a monoid morphism preserves the identity.
\end{proof}

\begin{proposition} \label{prop-monoid-invertible-image}
    Let $f \colon (M, \ast) \to (N, \cdot)$ be a monoid morphism. If $a \in M$ is invertible, then $f(a) \in N$ is also invertible.
\end{proposition}

\begin{proof}
    We have:
    \[
        f(a) \cdot f(a^{-1}) = f(a \ast a^{-1}) = f(e_{M}) = e_{N}.
    \]
    Similarly $f(a^{-1}) \cdot f(a) = e_{N}$. Hence, $f(a)$ is invertible and its inverse is $f(a^{-1})$.
\end{proof}

\begin{definition} \label{def-monoid-isomorphism}
    A \emph{monoid isomorphism} is a bijective monoid morphism.
\end{definition}

\begin{theorem} \label{thm-monoid-isomorphism-inverse}
    If $f$ is a monoid isomorphism, then the inverse map $f^{-1}$ is a monoid morphism. Moreover, it is a monoid isomorphism.
\end{theorem}

\begin{proof}
    By \ref{thm-magma-isomorphism-inverse} and the fact that $f^{-1}$ preserves also identity.
\end{proof}

\subsubsection{Groups}

\begin{definition} \label{def-group}
    A \emph{group} is a monoid such that any element is invertible.
\end{definition}

\begin{definition} \label{def-group-order}
    The \emph{order} of a group $G$, denoted by $\lvert G \rvert$ or $\# G$, is the cardinality of its underlying set. 
\end{definition}

\begin{definition} \label{def-group-commutative}
    A group is \emph{commutative} (or \emph{abelian}) if the underlying magma is commutative.
\end{definition}

\begin{proposition} \label{prop-group-product}
    Any product of groups is a group.
\end{proposition}

\begin{proof}
    It is induced by \ref{prop-monoid-product} and \ref{prop-magma-product-invertibility}.
\end{proof}

\subsection{Subgroups, group morphisms}
\subsubsection{Subgroups}

\begin{definition} \label{def-subgroup}
    Let $(G, \ast)$ be a group. A subset $H \subseteq G$ is a \emph{subgroup} of $G$ if it is a submonoid and is stable by inversion: 
    $$
        \forall x \in H, \; x^{-1} \in H.
    $$
\end{definition}

\begin{remark} \label{rem-group-canonical-subgroup}
    The subgroup $H$ induces the canonical group $(H, \ast_{\vert H \times H})$.
\end{remark}

\begin{proposition} \label{prop-subgroup-criterion}
    Let $(G, \ast)$ be a group. A non-empty subset $H \subseteq G$ is a subgroup of $G$ if and only if:
    \[
        \forall x, y \in H, \quad x \ast y^{-1} \in H.
    \]
\end{proposition}

\begin{proof}
    The direct way is a consequence of the definition of subgroups.

    Conversely, suppose $H \neq \emptyset$ and satisfies the given criterion. Since $H$ is non-empty, there exists some $x \in H$. Applying the criterion to $x$ and $x$ yields $x \ast x^{-1} = e \in H$. Now, for any $y \in H$, applying the criterion to $e$ and $y$ gives $e \ast y^{-1} = y^{-1} \in H$. Finally, for any $x, y \in H$, we know $y^{-1} \in H$, so applying the criterion to $x$ and $y^{-1}$ gives $x \ast (y^{-1})^{-1} = x \ast y \in H$.
\end{proof}

\begin{proposition} \label{prop-group-subgroups-intersection}
    Given a group $G$, any intersection of subgroups of $G$ is a subgroup of $G$.
\end{proposition}

\begin{proof}
    By \ref{rem-monoid-submonoids-intersection} and the fact that stability under inversion is preserved after intersection.
\end{proof}

\subsubsection{Group morphisms}

\begin{definition} \label{def-group-morphism}
    A \emph{group morphism} is a morphism of underlying magmas.
\end{definition}

\begin{notation} \label{not-group-operation}
    From now on, the group operation will be denoted multiplicatively. 
    Hence, for any $x, y \in G$, the product $x \ast y$ will be implicitly written as $xy$. 
    When the context requires it (notably for abelian groups), the additive notation $x + y$ will be preferred.
\end{notation}

\begin{definition} \label{def-group-morphism-domain}
    Let $f \colon G \to G^{\prime}$ be a group morphism. We say that $G$ is the \emph{domain} of $f$, and $G^{\prime}$ is the \emph{codomain} of $f$.
\end{definition}

\begin{proposition} \label{prop-group-morphism-reversibility}
    Let $G, G^{\prime}$ be two groups and $f \colon G \to G^{\prime}$ be a group morphism. Then $f(e_{G}) = e_{G^{\prime}}$ and for all $x \in G, \; f(x)^{-1} = f(x^{-1})$.
\end{proposition}

\begin{proof}
    By the property of group morphism, we have
    \[
        f(e_{G}) = f(e_{G} e_{G}) = f(e_{G}) f(e_{G})
    \]
    that implies $f(e_{G}) = e_{G^\prime}$ after multiplication by $f(e_{G})^{-1}$. Equality $f(x)^{-1} = f(x^{-1})$ follows from the property we have just shown.
\end{proof}

\begin{proposition} \label{prop-group-morphism-composition}
    A composition of group morphisms is a group morphism.
\end{proposition}

\begin{proof}
    Induced by \ref{prop-magma-morphism-composition}.
\end{proof}

\begin{theorem} \label{thm-group-morphism-image}
    Let $f \colon G \to G^{\prime}$ be a group morphism.
    \begin{enumerate}[label=(\roman*)]
        \item If $H$ is a subgroup of $G$, then $f(H)$ is a subgroup of $G^{\prime}$,
        \item If $H^{\prime}$ is a subgroup of $G^{\prime}$, then $f^{-1}(H^{\prime})$ is a subgroup of $G$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    Induced by \ref{thm-monoid-morphism-image} and \ref{prop-group-morphism-reversibility}.
\end{proof}

\begin{definition} \label{def-group-morphism-kernel}
    The \emph{kernel} of a group morphism $f \colon G \to G^{\prime}$ is defined by
    \[
        \Ker (f) = \lbrace a \in G \mid f(a) = e_{G^{\prime}} \rbrace
    \]
\end{definition}

\begin{proposition} \label{prop-group-morphism-kernel-subgroup}
    The kernel of a group morphism $f \colon G \to G^{\prime}$ is a subgroup of $G$.
\end{proposition}

\begin{proof}
    Induced by \ref{thm-group-morphism-image}-(ii).
\end{proof}

\begin{proposition} \label{prop-group-morphism-inj-criterion}
    A morphism of groups $f \colon G \to G^{\prime}$ is injective if and only if its kernel is the trivial subgroup $\lbrace e_{G} \rbrace$.
\end{proposition}

\begin{proof}
    The if condition: Let $x, x^{\prime} \in G$ be such that $f(x) = f(x^{\prime})$, which is equivalent to $f(x) f(x^{\prime})^{-1} = e_{G^{\prime}}$. However, $f(x) f(x^{\prime})^{-1} = f(x) f((x^{\prime})^{-1}) = f(x (x^{\prime})^{-1})$. Which implies, by hypothesis $x (x^{\prime})^{-1} = e_{G}$. So $x = x^{\prime}$. The converse way is obvious.
\end{proof}

\begin{definition} \label{def-group-morphism-image}
    The \emph{image} of a group morphism is the set image of the underlying map.
\end{definition}

\begin{proposition} \label{prop-group-morphism-image-subgroup}
    The image of a group morphism $f \colon G \to G^{\prime}$ is a subgroup of $G^{\prime}$.
\end{proposition}

\begin{proof}
    Induced by \ref{thm-group-morphism-image}-(i).
\end{proof}

\subsubsection{Isomorphism of groups}

\begin{definition} \label{def-group-isomorphism}
    A \emph{group isomorphism} is a bijective group morphism.
\end{definition}

\begin{notation}
    If $f \colon G \to G^{\prime}$ is an isomorphism, then we say that $G$ is \emph{isomorphic} to $G^{\prime}$. If $G = G^{\prime}$, we say that $f$ is an \emph{automorphism}.
\end{notation}

\begin{theorem} \label{thm-group-isomorphism-composition}
    The composition of two group isomorphisms is a group isomorphism.
\end{theorem}

\begin{proof}
    The composition of two bijective maps is a bijective map.
\end{proof}

\begin{theorem} \label{thm-group-isomorphism-inverse}
    If $f$ is a group isomorphism, then the inverse $f^{-1}$ is a group morphism. Moreover, it is a group isomorphism.
\end{theorem}

\begin{proof}
    Induced by \ref{thm-magma-isomorphism-inverse}.
\end{proof}

\begin{proposition}
    The relation "is isomorphic to" is an equivalence relation on the class of groups. Its equivalence classes are called \emph{isomorphism classes}.
\end{proposition}

\begin{proof}
    Induced by \ref{thm-group-isomorphism-inverse} and \ref{thm-group-isomorphism-composition}.
\end{proof}

\subsection{Normal subgroup and quotient}

\begin{definition} \label{def-group-normal-subgroup}
    Let $G$ be a group. A subgroup $H$ of $G$ is said to be a \emph{normal} subgroup if:
    \[
        \forall h \in H, \forall g \in G, \; g h g^{-1} \in H
    \]
\end{definition}

\begin{remark}
    If $G$ is an abelian group, then any subgroup of $G$ is a normal subgroup.
\end{remark}

\begin{definition} \label{def-group-canonical-surjection}
    Let $G$ be a group and $H$ a subgroup of $G$. For $g \in G$, we denote the subset $gH = \lbrace gh \mid h \in H \rbrace$, called the \emph{left coset} of $g$ modulo $H$. The left cosets constitute a partition of $G$; the set of left cosets modulo $H$ is called the \emph{left quotient} of $G$ by $H$, denoted by $G / H$; the map
    \[
        \begin{aligned}
            \pi \colon G &\longrightarrow G / H \\
            g &\longmapsto gH
        \end{aligned}
    \]
    is called the \emph{canonical surjection}.
\end{definition}

\begin{definition} \label{def-group-subgroup-index}
    Let $G$ be a group and $H$ a subgroup of $G$. The \emph{index} of $H$ in $G$, denoted by $[G : H]$, is the cardinality of the quotient $G / H$.
\end{definition}

\begin{proposition}
    Let $G$ be a group and $H$ a subgroup of $G$. Then we have:
    \[
        \lvert G \rvert = [G : H] \lvert H \rvert
    \]
\end{proposition}

\begin{proof}
    Let $X$ be a subset of $G$ such that the restriction of the canonical surjection $\pi \colon G \to G / H$ to $X$ is bijective. Then we have $[G : H] = \text{card}(X)$ ($X$ is a system of representatives of left cosets of $G$ by $H$).
    Let the map
    \[
        \begin{aligned}
            f \colon X \times H &\longrightarrow G \\
            (x, h) &\longmapsto xh
        \end{aligned}
    \]
    which is a bijection. Hence, $\lvert G \rvert = \text{card} (X \times H) = [G : H] \lvert H \rvert$.
\end{proof}

\begin{remark} \label{rem-group-subgroup-order-divisibility}
    If the order of $G$ is finite, then the order of $H$ divides the order of $G$.
\end{remark}

\begin{proposition} \label{prop-group-quotient}
    Let $G$ be a group and $H$ a subgroup of $G$. The following conditions are equivalent:
    \begin{enumerate}[label=(\roman*)]
        \item $H$ is a normal subgroup,
        \item There exists a group structure on the quotient $G / H$, such that the canonical surjection $G \to G / H$ is a morphism of groups,
        \item There exists a group $G^{\prime}$ and a morphism of groups $f \colon G \to G^{\prime}$ such that $H$ is the kernel of $f$.
    \end{enumerate}
\end{proposition}

\begin{proof}
    Omitted.
\end{proof}

\subsection{Generalities}
\subsubsection{Rings and subrings}

\begin{definition} \label{def-ring}
    A \emph{ring} is a triple $(A, +, \cdot)$ such that $(A, +)$ is an abelian group, $(A, \cdot)$ is a monoid, and the operation $\cdot$ is distributive over $+$:
    \[
        \forall a, b, c \in A, \quad a \cdot (b + c) = a \cdot b + a \cdot c \quad \text{and} \quad (b + c) \cdot a = b \cdot a + c \cdot a
    \]
\end{definition}

\begin{notation} \label{not-ring-identities}
    The identity elements are denoted by: $0$ for $(A, +)$ and $1$ for $(A, \cdot)$.
\end{notation}

\begin{notation} \label{not-ring}
    Most often, a ring $(A, +, \cdot)$ is simply denoted by $A$.
\end{notation}

\begin{remark}
    Let $A$ be a ring. Then for all $a \in A$:
    \[
        0 \cdot a = a \cdot 0 = 0 \quad \text{and} \quad (-1) \cdot a = -a = a \cdot (-1)
    \]
\end{remark}

\begin{definition} \label{def-nonzero-ring}
    A ring $A$ is called \emph{nonzero} if it is not the trivial ring $\lbrace O_{A} \rbrace$.
\end{definition}

\begin{remark} \label{rem-ring-trivial-ring}
    In a ring $A$, we have $0_{A} = 1_{A}$ if and only if $A = \lbrace 0_{A} \rbrace$.
\end{remark}

\begin{definition} \label{def-ring-invertible}
    An element of a ring $A$ is \emph{invertible} if it is invertible in $(A, \cdot)$. We denote by $A^{\ast}$ the set of invertible elements of $A$.
\end{definition}

\begin{definition} \label{def-ring-field}
    A \emph{field} is a non-zero commutative ring $A$, such that $A^{\ast} = A \setminus \lbrace 0 \rbrace$.
\end{definition}

\begin{definition} \label{def-ring-subring}
    Let $A$ be a ring. A subset $B \subset A$ is a \emph{subring} of $A$ if it is a subgroup of $(A, +)$ and submonoid of $(A, \cdot)$.
\end{definition}

\begin{remark}
    A subring $B \subset A$ induces the canonical ring $(B, +_{\vert B \times B}, \cdot_{\vert B \times B})$.
\end{remark}

\begin{proposition} \label{prop-ring-generated}
    Let $A$ be a ring. The intersection of any family of subrings of $A$ is also a subring of $A$. In particular, if $X$ is a subset of $A$, the intersection of subrings of $A$ containing $X$ is the smallest subring of $A$ containing $X$, called the \emph{subring} of $A$ \emph{generated} by $X$.
\end{proposition}

\subsubsection{Ring morphisms}

\begin{definition} \label{def-ring-morphism}
    Let $A$ and $B$ be two rings. A map $f \colon A \to B$ is a \emph{ring morphism} if it is a group morphism from $(A, +)$ to $(B, +)$ and a monoid morphism from $(A, \cdot)$ to $(B, \cdot)$.
\end{definition}

\begin{proposition} \label{prop-ring-morphism-composition}
    A composition of ring morphisms is a ring morphism.
\end{proposition}

\begin{proof}
    By \ref{prop-group-morphism-composition} and \ref{prop-monoid-morphism-composition}.
\end{proof}

\begin{definition} \label{def-ring-isomorphism}
    An \emph{isomorphism of rings} is a bijective ring morphism.
\end{definition}

\begin{theorem} \label{thm-ring-isomorphism-inverse}
    If $f$ is a ring isomorphism, then the inverse $f^{-1}$ is a ring morphism. Moreover, it is a ring isomorphism.
\end{theorem}

\begin{proof}
    Induced by \ref{thm-group-isomorphism-inverse} and \ref{thm-monoid-isomorphism-inverse}.
\end{proof}

\begin{proposition} \label{prop-ring-invertible-image}
    Let $f \colon A \to B$ be a ring morphism. Then $f(A^{\ast}) \subset B^{\ast}$ and $f$ induces a group morphism $(A^{\ast}, \cdot) \to (B^{\ast}, \cdot)$.
\end{proposition}

\begin{proof}
    Induced by \ref{prop-monoid-invertible-image}.
\end{proof}

\begin{proposition} \label{prop-field-morphism-injectivity}
    Any morphism of fields is injective.
\end{proposition}

\begin{proof}
    Let $f \colon F \to G$ be a morphism between fields. We have $f(F \setminus \lbrace 0 \rbrace) \subset G \setminus \lbrace 0 \rbrace$ by \ref{prop-ring-invertible-image}, so $f^{-1}(\lbrace 0 \rbrace) \subset \lbrace 0 \rbrace$ which implies that $f$ is injective by \ref{prop-group-morphism-inj-criterion}.
\end{proof}

\subsubsection{Kernel and Image}

\begin{definition} \label{def-ring-morphism-kernel}
    The \emph{kernel} of a ring morphism is its kernel as an additive group morphism.
\end{definition}

\begin{proposition} \label{prop-ring-morphism-inj-criterion}
    A morphism of rings $f$ is injective if and only if $\Ker (f) = \lbrace 0 \rbrace$.
\end{proposition}

\begin{proof}
    Direct consequence of \ref{prop-group-morphism-inj-criterion}.
\end{proof}

\begin{definition} \label{def-ring-morphism-image}
    The \emph{image} of a ring morphism is the set image of the underlying map.
\end{definition}

\begin{proposition} \label{prop-ring-morphism-image-subring}
    The image of a ring morphism $f \colon A \to B$ is a subring of $B$.
\end{proposition}

\begin{proof}
    Induced by \ref{prop-group-morphism-image-subgroup} and \ref{thm-monoid-morphism-image}-(i).
\end{proof}

\subsection{Ideals and quotients}
\subsubsection{Ideals}

\begin{definition} \label{def-ring-ideal}
    Let $A$ be a ring. A subset $I \subset A$ is a \emph{left ideal} (resp. \emph{right ideal}) if it is a subgroup of $(A, +)$ and is stable under external multiplication:
    \[
        \forall a \in A, \forall b \in I, \; a \cdot b \in I \; (\text{resp. } b \cdot a \in I).
    \]
\end{definition}

\begin{definition} \label{def-ring-ideal-two-sided}
    A \emph{two-sided ideal} (or simply \emph{ideal}) is a subset $I \subset A$ that is both a left ideal and a right ideal.
\end{definition}

\begin{remark}
    In a commutative ring, being a left ideal is equivalent to being a right ideal.
\end{remark}

\begin{proposition} \label{prop-ring-morphism-kernel-ideal}
    The kernel of a ring morphism $f \colon A \to B$ is an ideal of $A$.
\end{proposition}

\begin{proof}
    By \ref{prop-group-morphism-kernel-subgroup} we have that it is a subgroup of $(A, +)$. Moreover, for all $a \in A$ and $b \in \Ker (f)$, we have $f(ab) = f(a) f(b) = 0$ (resp. $f(ba) = 0$), hence $ab$ (resp. $ba$) is in $\Ker (f)$.
\end{proof}

\begin{proposition} \label{prop-ring-ideal-intersection}
    Any intersection of ideals of a commutative ring is an ideal.
\end{proposition}

\begin{proof}
    The intersection is an additive subgroup by \ref{prop-group-subgroups-intersection} and the stability under external multiplication is preserved after intersection.
\end{proof}

\begin{definition} \label{def-ring-ideal-generated}
    Let $A$ be a commutative ring and a subset $X \subset A$. We denote by $\langle X \rangle \subset A$ the intersection of all ideals containing $X$. $\langle X \rangle$ is called the \emph{ideal generated} by $X$.
\end{definition}

\begin{proposition} \label{prop-ring-ideal-generated-smallest}
    Let $A$ be a commutative ring and a subset $X \subset A$. The generated ideal $\langle X \rangle$ is the smallest ideal containing $X$.
\end{proposition}

\begin{proof}
    Let $I$ be an ideal containing $X$. Then $\langle X \rangle \subset I$ because $I$ appears in the decomposition of $\langle X \rangle$ as an intersection of ideals containing $X$.
\end{proof}

\begin{proposition} \label{prop-ring-ideal-generated-eq}
    Let $A$ be a commutative ring and a subset $X \subset A$. Then we have the following equality
    \[
        \langle X \rangle = \Big\lbrace \sum_{i = 1}^n a_i x_i \mid n \in \mathbb{N}, \; (a_i) \in A^n, \; (x_i) \in X^n \Big\rbrace.
    \]
\end{proposition}

\begin{proof}
    Omitted.
\end{proof}

\begin{notation} \label{not-ring-ideal-generated-by-family}
    Let $A$ be a commutative ring and $(a_i)_{i \in I}$ a family of elements of $A$. We denote by $\big( (a_i)_{i \in I} \big)$ the ideal generated by the subset $\lbrace a_i, \; i \in I \rbrace$.
\end{notation}

\begin{definition} \label{def-ring-ideal-finitely-generated}
    A \emph{finitely generated ideal} is an ideal generated by a finite family.
\end{definition}

\begin{definition} \label{def-ring-principal-ideal}
    Let $A$ be a commutative ring. A \emph{principal ideal} of $A$ is an ideal generated by one element $a \in A$. We denote this ideal by $(a) \subset A$.
\end{definition}

\begin{proposition} \label{prop-ring-principal-ideals-inclusion}
    Let $A$ be a commutative ring and $a, b \in A$. We have the following equivalence:
    \[
        a \in (b) \iff (a) \subset (b).
    \]  
\end{proposition}

\begin{proof}
    If $a \in (b)$: since $(a)$ is the smallest ideal containing $a$, we have necessarily $(a) \subset (b)$.

    If $(a) \subset (b)$: since $a \in (a)$, hence $a \in (b)$.
\end{proof}

\begin{definition} \label{def-ring-ideals-sum}
    Let $A$ be a commutative ring and $I, J \subset A$ be two ideals. The \emph{ideal sum} of $I$ and $J$, denoted by $I + J$, is the ideal generated by $I \cup J$.
\end{definition}

\begin{definition} \label{def-ring-ideals-product}
    Let $A$ be a commutative ring and $I, J \subset A$ be two ideals. The \emph{ideal product} of $I$ and $J$, denoted by $IJ$, is the ideal generated by the set $\lbrace xy \mid x \in I, \; y \in J \rbrace$.
\end{definition}

\subsubsection{Quotients}

\begin{proposition}
    Let $A$ be a ring and an additive subgroup $I \subset A$. The following assertions are equivalent:
    \begin{enumerate}[label=(\roman*)]
        \item $I$ is an ideal of $A$,
        \item The group quotient $A / I$ has a ring structure, such that the canonical surjection $\pi \colon A \longrightarrow A / I$ is a ring morphism.
    \end{enumerate}
\end{proposition}

\begin{proof}
    If there exists a ring structure on $A / I$ such that the canonical surjection is a ring morphism, then its kernel, which is $I$, is an ideal of $A$ by \ref{prop-ring-morphism-kernel-ideal}. 
    
    Conversely, if $I$ is an ideal of $A$: We define the operation $\cdot$ on $A / I$ by:
    \[
        \forall a, b \in A, \; \pi(a) \cdot \pi(b) = \pi(ab)
    \]
    To show that this definition is correct, it requires to show that $\pi(ab) = \pi(a^{\prime} b^{\prime})$ for all $(a, b), (a^{\prime}, b^{\prime}) \in A^2$ such that $\pi(a) = \pi(a^{\prime})$ and $\pi(b) = \pi(b^{\prime})$. However, $\pi(x) = \pi(y) \iff x - y \in I$. Hence, it suffices to show that $ab - a^{\prime} b^{\prime} \in I$. To see it: $ab - a^{\prime} b^{\prime} = a(b - b^{\prime}) + (a - a^{\prime})b^{\prime} \in I$. To finish, the surjectivity of $\pi$ implies that this operation is well-defined on all $A / I$.
\end{proof}

\begin{definition} \label{def-ring-quotient}
    Let $A$ be a ring and $I \subset A$ an ideal. The ring quotient $A / I$ is called \emph{quotient of} $A$ \emph{by} $I$.
\end{definition}

\begin{theorem} \label{thm-ring-morphism-quotient}
    Let $f \colon A \to B$ be a ring morphism, an ideal $I \subset A$, and the canonical surjection $\pi \colon A \to A / I$. Then, the following assertions are equivalent:
    \begin{enumerate}[label=(\roman*)]
        \item $I \subset \Ker f$,
        \item There exists a ring morphism $g \colon A / I \to B$, such that $g \circ \pi = f$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    If $I \subset \Ker f$: we construct the map $g \colon A / I \longrightarrow B$ by setting $g(\pi(x)) = f(x)$ for all $x \in A$. 
        
    First, we must ensure that $g$ is well-defined. Let $x, x^{\prime} \in A$ such that $\pi(x) = \pi(x^{\prime})$. This equality implies that $x - x^{\prime} \in I$, therefore $x - x^{\prime} \in \Ker f$, meaning that $f(x - x^{\prime}) = f(x) - f(x^{\prime}) = 0_B$, hence $f(x) = f(x^{\prime})$. Thus, the value of $g$ does not depend on the choice of representative. It is easily proven that $g$ is a ring morphism.

    Conversely, assume there exists a ring morphism $g \colon A / I \longrightarrow B$ such that $g \circ \pi = f$. Let $x \in I$. By definition of the canonical surjection, we have $\pi(x) = 0$. Evaluating $f$ at $x$ yields:
    \[
        f(x) = (g \circ \pi)(x) = g(\pi(x)) = g(0) = 0.
    \]
    which proves that $I \subset \Ker f$.
\end{proof}

\begin{remark}
    A morphism $g$ as in \ref{thm-ring-morphism-quotient} is unique, due to the surjectivity of $\pi$
\end{remark}

\begin{corollary} \label{cor-ring-morphism-quotient}
Let $f \colon A \to B$ be a ring morphism. There exists a unique ring isomorphism:
    \[
        \tilde{f} \colon A / \Ker f \longrightarrow \Image f
    \]
    such that the following diagram is commutative:
    \[
        \begin{tikzcd}
            A \arrow[r, "f"] \arrow[d, "\pi"'] & B \\
            A / \Ker f \arrow[r, "\tilde{f}"'] & \Image f \arrow[u, hook]
        \end{tikzcd}
    \]
\end{corollary}

\begin{proof}
    Direct consequence of \ref{thm-ring-morphism-quotient}.
\end{proof}

\subsection{Nilpotency}

\begin{definition} \label{def-ring-nilpotent-element}
    An element $x$ of a ring $A$ is called \emph{nilpotent} if there exists a positive integer $n$, such that $x^n = 0_{A}$.
\end{definition}

\begin{definition} \label{def-ring-nilpotent-subset}
    Let $A$ be a ring. We denote $\Nil (A)$, the set of nilpotent elements of $A$.
\end{definition}

\begin{definition} \label{def-ring-nilpotency-index}
    The \emph{nilpotency index} (or \emph{nilpotency degree}) of a nilpotent element $x \in A$ is the smallest positive integer $n$, such that $x^n = 0_A$.
\end{definition}

\begin{lemma} \label{lem-ring-newton-formula}
    If $R$ is a commutative ring, then we have the Newton Formula:
    \[
        (x + y)^n = \sum\limits_{k = 0}^n \binom{n}{k} x^k y^{n - k}, \quad \forall x, y \in R, \; \forall n \in \mathbb{N}.
    \]
\end{lemma}

\begin{proof}
    The proof is obtained by induction.
\end{proof}

\begin{proposition} \label{prop-ring-nilpotent-ideal}
    If $R$ is a commutative ring, then $\Nil (R)$ is an ideal of $R$.
\end{proposition}

\begin{proof}
    Since $0_R$ is nilpotent, $\Nil (R)$ is non-empty. Let $x, y \in \Nil (R)$ with $n_x$ and $n_y$ their respective nilpotency indices. By \ref{lem-ring-newton-formula}, we have:
    \[
        \begin{aligned}
            (x - y)^{n_x + n_y} &= \sum\limits_{k = 0}^{n_x + n_y} \binom{n_x + n_y}{k} x^k (-y)^{n_x + n_y - k} \\
                &= \sum\limits_{k = 0}^{n_x} \binom{n_x + n_y}{k} x^k (-y)^{n_x + n_y - k} + \sum\limits_{k = n_x + 1}^{n_x + n_y} \binom{n_x + n_y}{k} x^k (-y)^{n_x + n_y - k}.
        \end{aligned}
    \]
    Since $(-y)^{n_x + n_y - k} = \pm y^{n_x + n_y - k} = 0$ for all $k \in \lbrace 0, \dots, n_x \rbrace$ and $x^k = 0$ for all $k \in \lbrace n_x + 1, \dots, n_x + n_y \rbrace$, we have $(x - y)^{n_x + n_y} = 0$. Therefore $x - y \in \Nil (R)$ and $\Nil (R)$ is an additive subgroup of $R$ by \ref{prop-subgroup-criterion}.

    Let $a \in R$ and $x \in \Nil (R)$ with nilpotency index $n_x$. By commutativity we have $(ax)^{n_x} = a^{n_x} x^{n_x} = 0$. Therefore $ax$ is nilpotent. Since $R$ is commutative, $\Nil (R)$ is a two-sided ideal of $R$.
\end{proof}

\subsection{Zero divisors, Integral domain}
\subsubsection{Zero divisors}

\begin{definition} \label{def-ring-lr-zero-div}
    An element $a$ of a ring $A$ is called a \emph{left zero divisor} (resp. \emph{right zero divisor}) if there exists a nonzero element $b \in A$, such that $ab = 0$ (resp. $ba = 0$).
\end{definition}

\begin{proposition} \label{prop-ring-lr-zero-div-eq}
    An element $a$ of a ring $A$ is a left zero divisor (resp. right zero divisor) if and only if $b \mapsto ab$ (resp. $b \mapsto ba$) is not injective.
\end{proposition}

\begin{proof}
    Obvious.
\end{proof}

\begin{definition} \label{def-ring-zero-div}
    An element of a ring that is both a left and a right zero divisor is called a \emph{zero divisor}.
\end{definition}

\begin{remark} \label{rem-ring-commutative-zero-div}
    In a commutative ring, any element is a left zero divisor if and only if it is a right zero divisor.
\end{remark}

\subsubsection{Integral domain}

\begin{definition} \label{def-integral-domain}
    An \emph{integral domain} is a nonzero commutative ring in which the product of any two nonzero elements is nonzero.
\end{definition}

\begin{proposition} \label{prop-integral-domain-contrapositive}
    If $R$ is an integral domain, then for all $a, b \in R$:
    \[
        ab = 0 \implies a = 0 \; \text{or} \; b = 0
    \]
\end{proposition}

\begin{proof}
    By contrapositive of the property of the definition of integral domain \ref{def-integral-domain}.
\end{proof}

\begin{remark}
    In particular, any nonzero subring of an integral domain is an integral domain.
\end{remark}

\begin{proposition} \label{prop-integral-domain-eq}
    A commutative nonzero ring is an \emph{integral domain} if and only if it has no nonzero zero divisors.
\end{proposition}

\begin{proof}
    Direct consequence of \ref{prop-integral-domain-contrapositive}.
\end{proof}

\begin{lemma} \label{lem-ring-cancellation-property}
    If $R$ is an integral domain and $a, b, c \in R$, then $a \neq 0, \; ab = ac$ implies $b = c$.
\end{lemma}

\begin{proof}
    $ab = ac$ implies $a (b - c) = 0$. However, $a \neq 0$ and $R$ is an integral domain, hence $b - c = 0$ by \ref{prop-integral-domain-contrapositive}.
\end{proof}

\begin{proposition} \label{prop-field-integral-domain}
    Any field is an integral domain.
\end{proposition}

\begin{proof}
    The definition of fields (see \ref{def-ring-field}) means that, the subset of nonzero elements is a multiplicative group. Hence any product of nonzero elements is a nonzero element.
\end{proof}

\subsection{Prime and maximal ideals}
\subsubsection{Prime ideals}

\begin{definition} \label{def-ring-prime-ideal}
    A \emph{prime ideal} of a commutative ring $R$ is an ideal $I \subset R$, such that the quotient ring $R / I$ is an integral domain.
\end{definition}

\begin{proposition} \label{prop-ring-prime-ideal-eq}
    $I \subset R$ is a prime ideal if and only if $I \neq R$ and for all $a, b \in R$:
    \[
        ab \in I \implies a \in I \; \text{or} \; b \in I.
    \]
\end{proposition}

\begin{proof}
    Direct consequence of \ref{prop-integral-domain-contrapositive}.
\end{proof}

\begin{proposition} \label{prop-ring-prime-ideals-intersection}
    Let $R$ be a commutative ring. The intersection of prime ideals of $R$ is equal to the ideal $\Nil (R)$.
\end{proposition}

\begin{proof}
    Let $x \in R$ be nilpotent and $I \subset R$ be a prime ideal. The image of $x$ in $R / I$ is also nilpotent. However $R / I$ is an integral domain, hence the image of $x$ is zero, which implies that $x \in I$.
    
    The converse way is omitted.
\end{proof}

\subsubsection{Maximal ideals}

\begin{definition} \label{def-ring-maximal-ideal}
    A \emph{maximal ideal} of a commutative ring $R$ is an ideal $I \subset R$, such that the quotient ring $R / I$ is a field.
\end{definition}

\begin{proposition} \label{prop-ring-maximal-ideal-eq}
    $I \subset R$ is a maximal ideal if and only if $I \neq R$ and for any ideal $J$:
    \[
        I \subset J \implies J = I \; \text{or} \; J = R.
    \]
\end{proposition}

\begin{proof}
    Omitted.
\end{proof}

\begin{lemma} \label{lem-union-totally-ordered-ideals}
    Let $R$ be a commutative ring and $T$ be a non-empty set of ideals of $R$. If $T$ is totally ordered with respect to inclusion, then $\mathfrak{a} = \bigcup_{I \in T} I$ is an ideal of $R$.
\end{lemma}

\begin{proof}
    Since $T$ is non-empty, hence $\mathfrak{a}$ is also non-empty. Let $x, y \in \mathfrak{a}$; there exists $I, J \in T$ such that $x \in I$ and $y \in J$. Since $T$ is totally ordered, we have $I \subset J$ or $J \subset I$, hence $x - y \in I$ or $x - y \in J$. So, $x - y \in \mathfrak{a}$ and $\mathfrak{a}$ is an additive subgroup by \ref{prop-subgroup-criterion}. Let $a \in R$ and $x \in \mathfrak{a}$; there exists $I \in T$ such that $x \in I$. Since $I$ is an ideal, hence $a x \in I$ and then $a x \in \mathfrak{a}$.
\end{proof}

\begin{lemma} \label{lem-prime-ideal-avoiding-multiplicative-subset}
    Let $R$ be a commutative ring and $S \subset R$ a multiplicative subset (see Definition~\ref{def-ring-multiplicative-subset}). Let $I \subset R$ be an ideal such that $I \cap S = \emptyset$, and let $\mathcal{T}$ be the set of ideals containing $I$ that are disjoint from $S$. Then:
    \begin{enumerate}[label=(\roman*)]
        \item $\mathcal{T}$ has a maximal element with respect to inclusion.
        \item Any maximal element of $\mathcal{T}$ is a prime ideal of $R$.
    \end{enumerate}
\end{lemma}

\begin{proof}
    To show this, we use the Zorn's lemma on $\mathcal{T}$. To start, we ensure that the hypotheses of the lemma are satisfied. Let $T \subset \mathcal{T}$ be a subset totally ordered; we prove that $T$ has an upper bound (in $\mathcal{T}$). If $T = \emptyset$, then $I$ is an upper bound for $T$ since $I$ is a minimal element of $\mathcal{T}$. If $T$ is non-empty; let $\mathfrak{a} = \bigcup_{J \in T} J$ which is an ideal of $R$ by \ref{lem-union-totally-ordered-ideals}. Moreover, $\mathfrak{a}$ is disjoint from $S$ (and it contains $I$), hence $\mathfrak{a} \in \mathcal{T}$. From definition of $\mathfrak{a}$, we have for all $\mathfrak{b} \in \mathcal{T}$:
    \[
        \mathfrak{a} \subset \mathfrak{b} \iff \forall J \in T, J \subset \mathfrak{b},
    \]
    hence $\mathfrak{a}$ is the upper bound of $T$ in $\mathcal{T}$.

    We now apply Zorn's lemma to $\mathcal{T}$ and we obtain (i). Let's prove (ii); Let $\mathfrak{p}$ be a maximal element of $\mathcal{T}$. Since $\mathfrak{p} \cap S = \emptyset$, hence $1 \not \in \mathfrak{p}$ and so $\mathfrak{p} \neq R$. Let $x, y \in R \setminus \mathfrak{p}$; Since $\mathfrak{p}$ is a maximal element of $\mathcal{T}$, hence the sum ideal $\mathfrak{p} + (x)$ is not an element of $\mathcal{T}$. Moreover, since $\mathfrak{p} + (x)$ contains $I$, we have $(\mathfrak{p} + (x)) \cap S \neq \emptyset$; There exists $s \in S, \; u \in \mathfrak{p}$ and $a \in R$ such that $s = u + a x$. In the same way, there exists $t \in S, \; v \in \mathfrak{p}$ and $b \in R$ such that $t = v + b y$. We have:
    \[
        s t = u v + a x v + b y u + a b x y
    \]
    If $x y \in \mathfrak{p}$, then $s t \in \mathfrak{p}$. Since $s t \in S$, this implies $\mathfrak{p} \cap S \neq \emptyset$, which is contradictory. We have just shown that:
    \[
        x, y \in R \setminus \mathfrak{p} \implies x y \in R \setminus \mathfrak{p}
    \]
\end{proof}

\begin{proposition} \label{prop-ideal-included-in-maximal-ideal}
    Let $R$ be a commutative ring and an ideal $I \subset R$ such that $I \neq R$. Then there exists a maximal ideal containing $I$.
\end{proposition}

\begin{proof}
    By hypothesis $I \neq R$, hence it is disjoint from $S = \lbrace 1 \rbrace$. We use \ref{lem-prime-ideal-avoiding-multiplicative-subset}; So the set of ideals containing $I$ and being disjoint from $S$ has a maximal element, denoted by $J$. Moreover, by \ref{prop-ring-maximal-ideal-eq} we deduce that $J$ is a maximal ideal.
\end{proof}

\begin{corollary} \label{cor-invertible-element-in-maximal-ideal}
    Let $R$ be a commutative ring. The set $R \setminus R^{\ast}$ of non-invertible elements is equal to the union of maximal ideals of $R$.
\end{corollary}

\begin{proof}
    Since $x \in R^{\ast} \iff (x) = R$, hence $x \in R \setminus R^{\ast} \iff (x) \neq R$. By \ref{prop-ideal-included-in-maximal-ideal}, we have $(x) \neq R \iff (x)$ is included in a maximal ideal of $R$.
\end{proof}

\subsection{Localization, Field of fractions}
\subsubsection{Localization}

\begin{definition} \label{def-ring-multiplicative-subset}
    A subset $S$ of a ring $A$ is a \emph{multiplicative subset} if it contains $1_A$ and is closed under multiplication:
    \[
        \forall a, b \in S, \; ab \in S.
    \]
\end{definition}

\begin{remark}
    If $R$ is an integral domain, then $R \setminus \lbrace 0 \rbrace$ is a multiplicative subset.
\end{remark}

\begin{proposition} \label{prop-ring-localization-relation}
    Let $R$ be a nonzero commutative ring and $S \subset R$ a multiplicative subset. The relation $\mathcal{R}_S$, on $R \times S$, defined as follows:
    \[
        (x, s) \sim (x^{\prime}, s^{\prime}) \iff \exists \sigma \in S, \; \sigma x s^{\prime} = \sigma x^{\prime} s.
    \]
    is an equivalence relation.
\end{proposition}

\begin{proof}
    It is clear that this relation is reflexive and symmetric. Let $(x, s), (x^{\prime}, s^{\prime}), (x^{\prime \prime}, s^{\prime \prime}) \in R \times S$, such that $\sigma x s^{\prime} = \sigma x^{\prime} s$ and $\sigma^{\prime} x^{\prime} s^{\prime \prime} = \sigma^{\prime} x^{\prime \prime} s^{\prime}$, with $\sigma, \sigma^{\prime} \in S$. Multiplying the first equation by $\sigma^{\prime} s^{\prime \prime}$, and the second by $\sigma s$, we obtain
    \[
        \sigma \sigma^{\prime} s^{\prime} x s^{\prime \prime} = \sigma \sigma^{\prime} x^{\prime} s s^{\prime \prime} = \sigma \sigma^{\prime} s^{\prime} x^{\prime \prime} s 
    \]
    However, $\sigma \sigma^{\prime} s^{\prime} \in S$. Hence, we have $(x, s) \sim (x^{\prime \prime}, s^{\prime \prime})$.
\end{proof}

\begin{definition} \label{def-ring-quotient-localization}
    Let $R$ be a nonzero commutative ring and $S \subset R$ a multiplicative subset. We denote by $S^{-1} R$, the quotient of $R \times S$ by the relation $\mathcal{R}_S$. The equivalence class of an element $(x, s) \in R \times S$ is denoted by the fraction $x / s$.
\end{definition}

\begin{proposition} \label{prop-ring-quotient-localization-operations}
    Let $R$ be a nonzero commutative ring and $S \subset R$ a multiplicative subset. Then, we can define two operations $+$ and $\cdot$ by:
    \[
        \frac{x}{s} + \frac{x^{\prime}}{s^{\prime}} = \frac{s^{\prime} x + s x^{\prime}}{s s^{\prime}} \; \text{and} \; \frac{x}{s} \cdot \frac{x^{\prime}}{s^{\prime}} = \frac{x x^{\prime}}{s s^{\prime}}
    \]
    In particular, these definitions are independent of the choice of representatives for the fraction classes.
\end{proposition}

\begin{proof}
    Omitted.
\end{proof}

\begin{proposition} \label{prop-ring-quotient-localization-ring}
    Let $R$ be a nonzero commutative ring and $S \subset R$ a multiplicative subset. Then, $S^{-1} R$ equipped with the operations defined in \ref{prop-ring-quotient-localization-operations} has a ring structure.
\end{proposition}

\begin{proof}
    It is clear that the elements $0 / 1$ and $1 / 1$ are respectively the identity elements of $+$ and $\cdot$. Through explicit calculations we demonstrate the associativity of these operations. To finish, the additive inverse of $x / s$ is $ (-x) / s$.
\end{proof}

\begin{remark}
    It is clear that the quotient ring $S^{-1} R$, as in \ref{prop-ring-quotient-localization-ring}, is a commutative ring.
\end{remark}

\begin{proposition} \label{prop-ring-localization-zero}
    Let $R$ be a nonzero commutative ring and $S \subset R$ a multiplicative subset. If $0 \in S$, then $S^{-1} R$ is the zero ring.
\end{proposition}

\begin{proof}
    If $0 \in S$, then any element $x / s \in S^{-1} R$, is equal to $0 / 1$. Therefore, $S^{-1} R$ is the trivial ring.
\end{proof}

\begin{proposition} \label{prop-ring-localization-morphism}
    Let $R$ be a nonzero commutative ring and $S \subset R$ a multiplicative subset. Then the following canonical map: 
    \[
        \begin{aligned}
            \eta \colon R &\longrightarrow S^{-1} R \\
            x &\longmapsto x / 1
        \end{aligned}
    \]
    is a ring morphism.
\end{proposition}

\begin{proof}
    Follows immediately from the definition of the $S^{-1} R$ operations in \ref{prop-ring-quotient-localization-operations}.
\end{proof}

\begin{proposition} \label{prop-ring-localization-morphism-injectivity}
    Let $R$ be a nonzero commutative ring and $S \subset R$ a multiplicative subset. Then the morphism $\eta$, defined as in \ref{prop-ring-localization-morphism}, is injective if and only if $S$ does not contain a zero divisor.
\end{proposition}

\begin{proof}
    Induced by the fact that $\Ker (\eta) = \lbrace x \in R \mid \exists \sigma \in S, \; \sigma x = 0 \rbrace$.
\end{proof}

\begin{proposition} \label{prop-ring-localization-eq-condition}
    Let $R$ be a nonzero commutative ring and $S \subset R$ a multiplicative subset. If $S$ does not contain a zero divisor, then:
    \[
        \frac{x}{s} = \frac{x^{\prime}}{s^{\prime}} \iff s^{\prime} x = s x^{\prime}
    \]
\end{proposition}

\begin{proof}
    If $x / s = x^{\prime} / s^{\prime}$, then there exists $\sigma \in S$ such that $\sigma (x s^{\prime} - x^{\prime} s) = 0$. However, $\sigma$ is not a zero divisor, hence $x s^{\prime} - x^{\prime} s = 0$. 
\end{proof}

\subsubsection{Field of fractions}

\begin{definition} \label{def-fraction-field}
    If $R$ is an integral domain, then we denote $\Frac R = (R \setminus \lbrace 0 \rbrace)^{-1} R$, called the \emph{field of fractions} of $R$.
\end{definition}

\begin{proposition} \label{prop-ring-fraction-field}
    If $R$ is an integral domain, then the field of fractions $\Frac R$ is a field.
\end{proposition}

\begin{proof}
    $R \setminus \lbrace 0 \rbrace$ does not contain a zero divisor. By \ref{prop-ring-localization-morphism-injectivity}, the canonical morphism $R \to \Frac R$ is injective. Hence, $\Frac R$ is a nonzero ring. Let $x / s \in \Frac R$ be a nonzero element ($x \neq 0$ by \ref{prop-ring-localization-eq-condition}), then its inverse is $s / x$. To show this, it suffices to verify that its product is equal to $1_{\Frac R}$.
\end{proof}

\begin{remark}
    $R$ is identified with a subring of $\Frac R$.
\end{remark}

\begin{proposition}
    Let $R$ be an integral domain and $S \subset R$ a multiplicative subset. If $S$ does not contain a zero divisor, then $S^{-1} R$ is identified with a subring of $\Frac R$.
\end{proposition}

\begin{proof}
    Omitted.
\end{proof}

\begin{proposition}
    Let $R$ be an integral domain. For any field $K$ and injective ring morphism $j \colon R \to K$, there exists a unique ring morphism $\Frac R \to K$ which extends $j$.
\end{proposition}

\begin{proof}
    Omitted.
\end{proof}

\subsection{Divisibility}

\begin{definition} \label{def-ring-lr-divisibility}
    Let $A$ be a ring and $a, b \in A$. $a$ is a \emph{left divisor} (resp. \emph{right divisor}) of $b$ if there exists $x \in A$ such that $ax = b$ (resp. $xa = b$).
\end{definition}

\begin{definition} \label{def-ring-divisibility}
    Let $A$ be a ring and $a, b \in A$. $a$ is a \emph{divisor} (or \emph{two-sided divisor}) of $b$ if it is both a left divisor and a right divisor of $b$. In this case, we denote $a \mid b$.
\end{definition}

\begin{remark}
    In a commutative ring, the notions of left divisor, right divisor, and two-sided divisor coincide.
\end{remark}

\begin{remark}
    According to this definition, every element is a divisor of $0$. Therefore, we will make an exception and definitively adopt the definition \ref{def-ring-zero-div} to define a divisor of zero. This clarification is made to avoid any conflict between the definitions.
\end{remark}

\begin{remark}
    An element $x \in R$ of a commutative ring is invertible if and only if $x \mid 1$.
\end{remark}

\begin{proposition} \label{prop-ring-divisibility-eq}
    Let $R$ be a commutative ring and $a, b \in R$. We have the following equivalences
    \[
        a \mid b \iff (b) \subset (a).
    \]
\end{proposition}

\begin{proof}
    If $a \mid b$: there exists $x \in R$ such that $b = ax$, therefore $b \in (a)$.

    If $b \in (a)$: by \ref{prop-ring-ideal-generated-eq}, there exists $x \in R$ such that $b = ax$, therefore $a \mid b$.

    The final result is induced by  \ref{prop-ring-principal-ideals-inclusion}.
\end{proof}

\begin{definition} \label{def-ring-associated-elements}
    Let $R$ be a commutative ring. Two elements $a, b \in R$ are \emph{associates} if $a \mid b$ and $b \mid a$.
\end{definition}

\begin{proposition} \label{prop-ring-associated-elements-eq}
    Let $R$ be an integral domain. Two nonzero elements $a, b \in R$ are associates if and only if there exists $u \in R^{\ast}$ such that $b = u a$.
\end{proposition}

\begin{proof}
    If $a$ and $b$ are associated: $a$ divides $b$, then there exists $u \in R$ such that $b = u a$. Conversely, $b$ divides $a$, so there exists $v \in R$ such that $a = v b$. We show that $u$ is invertible: 
    $b = u a = u (v b) = (u v) b$, and $R$ is an integral domain, therefore $u v = 1$ by \ref{lem-ring-cancellation-property}. Similarly $a = v b = v (u a) = (v u) a$, therefore $v u = 1$.

    Conversely, if there exists $u \in R^{\ast}$ such that $b = u a$, then $a = b u^{-1}$. Therefore $a \mid b$ and $b \mid a$.
\end{proof}

\subsection{Noetherian rings}

\begin{definition} \label{def-Noetherian-ring}
    A ring is \emph{Noetherian} if every increasing sequence (for inclusion) of its ideals $(I_n)_{n \in \mathbb{N}}$ is stationary.
\end{definition}

\begin{proposition} \label{prop-Noetherian-ring-eq}
    $A$ is a Noetherian ring if and only if every ideal of $A$ is finitely generated.
\end{proposition}

\begin{proof}
    If there exists an ideal $I \subset A$ not finitely generated: we can construct a sequence $(x_n)$ such that for all $n$, $x_n \in I$ and $x_{n + 1} \not \in (x_1, \dots, x_n)$. Therefore, if $I_n = (x_1, \dots, x_n)$ then $(I_n)$ is an increasing sequence not stationary and hence $A$ is not Noetherian.

    If every ideal of $A$ is finitely generated: Let $(I_n)$ be an increasing sequence of ideals and $I = \bigcup I_n$ which is an ideal also. By hypothesis, $I$ is finitely generated and we can write $I = (x_1, \dots, x_m)$. For all $i$, there exists $k_i$ such that $x_i \in I_{k_i}$. Let $N = \max_i k_i$, then we have for all $i$, $x_i \in I_N$. Therefore $I_N = I$.
\end{proof}

\section{Modules}
\subsection{Modules, morphisms and submodules}

\begin{definition} \label{def-lr-module}
    Let $A$ be a nonzero ring. A \emph{left $A$-module} (resp. \emph{right $A$-module}) $M$ consists of an abelian group $(M, +)$ and an operation
    \[
        \cdot \colon A \times M \to M
    \]
    such that for all $a, b \in A$ and $x, y \in M$ we have:
    \begin{enumerate}[label=(\roman*)]
        \item $a \cdot (x + y) = a \cdot x + a \cdot y$,
        \item $(a + b) \cdot x = a \cdot x + b \cdot x$,
        \item $(ab) \cdot x = a \cdot (b \cdot x)$ (resp. $(ba) \cdot x = a \cdot (b \cdot x)$),
        \item $1_{A} \cdot x = x$.
    \end{enumerate}
    The operation $\cdot$ is called \emph{scalar multiplication}. Often the symbol $\cdot$ is omitted. 
\end{definition}

\begin{notation}
    One may write $_{A} M$ (resp. $M_{A}$) to emphasize that $M$ is a left $A$-module (resp. right $A$-module).
\end{notation}

\begin{remark}
    For clarity, the scalar multiplication in a right module $M_{A}$ is written on the right: for $a \in A$ and $x \in M$, we denote $x \cdot a$.
\end{remark}

\begin{proposition}
    Let $R$ be a nonzero ring. If $R$ is commutative, then left $R$-modules are the same as right $R$-modules and are simply called \emph{$R$-modules}.
\end{proposition}

\begin{remark}
    If $K$ is a field, then the $K$-modules are exactly the $K$-vector spaces.
\end{remark}

\begin{remark}
    Each ring $A$ can be considered as a left $A$ module and a right $A$ module. For this, we define scalar multiplication by the ring multiplication.
\end{remark}

\begin{remark}
    The data for a $\mathbb{Z}$-module is equivalent to the data for an abelian group.
\end{remark}

\subsubsection{Module morphisms}

\begin{definition} \label{def-module-morphism}
    Let $A$ be a nonzero ring and $M, N$ be two left (resp. right) $A$-modules. A map $f \colon M \to N$ is called $A$-module morphism (or $A$-linear map) if it is additive group morphism such that for all $a \in A$ and $x \in M$, we have $f(a x) = a f(x)$ (resp. $f(xa) = f(x) a$). 
\end{definition}

\begin{proposition} \label{prop-module-morphism-composition}
    Let $A$ be a nonzero ring. The composition of morphisms of left (resp. right) $A$-modules is an $A$-module morphism.
\end{proposition}

\begin{proof}
    The composition is a group morphism by \ref{prop-group-morphism-composition}. The scalar multiplication property is clearly preserved.
\end{proof}

\begin{remark}[Identity morphism]
    Let $A$ be a nonzero ring and $M$ be a left (resp. right) module. The identity map of $M$ is an $A$-module morphism.
\end{remark}

\begin{definition} \label{def-module-isomorphism}
    Let $A$ be a nonzero ring. A morphism of left (resp. right) $A$-modules is called isomorphism if it is bijective.
\end{definition}

\begin{proposition} \label{prop-module-morphism-reversibility}
    Let $A$ be a nonzero ring and $M, N$ be two left (resp. right) $A$-modules. If $f \colon M \to N$ is an $A$-module isomorphism, then $f^{-1}$ is an $A$-module morphism. Moreover, it is an $A$-module isomorphism.
\end{proposition}

\begin{proof}
    Assume that $M, N$ are left modules, The right modules case is analogous. By \ref{thm-group-isomorphism-inverse}, $f^{-1} \colon N \to M$ is an additive group morphism. Let $a \in A$ and $y \in N$. By surjectivity of $f$, there exists $x \in M$ such that $y = f(x)$. We have $f^{-1}(a y) = f^{-1}(a f(x)) = f^{-1}(f(a x)) = a x = a f^{-1}(y)$. Hence, $f^{-1}$ is an $A$-module morphism.
\end{proof}

\begin{definition} \label{def-module-morphism-kernel-image}
    Let $A$ be a nonzero ring and $f$ be a morphism of left (resp. right) $A$-modules.
    \begin{enumerate}[label=(\roman*)]
        \item The kernel of $f$, denoted by $\Ker f$, is its kernel as a group morphism.
        \item The image of $f$, denoted by $\Image f$, is its image as a group morphism.
    \end{enumerate}
\end{definition}

\subsubsection{Submodules}

\begin{definition} \label{def-submodule}
    Let $A$ be a nonzero ring and $M$ be a left (resp. right) $A$-module. A submodule of $M$ is a additive subgroup $N \subset M$ such that for all $a \in A$ and $x \in N$, we have $ax \in N$ (resp. $xa \in N$).
\end{definition}

\begin{proposition} \label{prop-submodule-canonical-structure}
    Let $A$ be a nonzero ring and $M$ be a left (resp. right) $A$-module. A submodule of $M$ has a canonical structure of left (resp. right) $A$-module, induced by the module structure of $M$.
\end{proposition}

\begin{remark}[Submodules of a ring]
    Let $A$ be a nonzero ring, viewed as a left (resp. right) $A$-module. Then the submodules of $A$ are exactly the left (resp. right) ideals.
\end{remark}

\begin{proposition} \label{prop-module-morphism-kernel-submodule}
    Let $A$ be a nonzero ring, $M, N$ be two left (resp. right) $A$-modules and $f \colon M \to N$ be an $A$-module morphism. $\Ker f$ is a submodule of $M$.
\end{proposition}

\begin{proof}
    Assume that $M, N$ are left modules, The right modules case is analogous. By \ref{prop-group-morphism-kernel-subgroup}, $\Ker f$ is a subgroup of $M$. Let $a \in A$ and $x \in \Ker f$, we have $f(a x) = a f(x) = a \cdot 0 = 0$, hence $a x \in \Ker f$.
\end{proof}

\begin{proposition} \label{prop-module-morphism-image-submodule}
    Let $A$ be a nonzero ring, $M, N$ be two left (resp. right) $A$-modules and $f \colon M \to N$ be an $A$-module morphism. $\Image f$ is a submodule of $N$.
\end{proposition}

\begin{proof}
    Assume that $M, N$ are left modules, The right modules case is analogous. By \ref{prop-group-morphism-image-subgroup}, $\Image f$ is an subgroup of $N$. Let $a \in A$ and $y \in \Image f$. There exists $x \in M$ such that $y = f(x)$. We have $a y = a f(x) = f(a x) \in \Image f$.
\end{proof}

\subsubsection{Ring change}

\begin{proposition} \label{prop-module-ring-change}
    Let $A, B$ be two nonzero rings and $M$ be a left (resp. right) $B$-module. If $\eta \colon A \to B$ is a ring morphism, then $M$ has a left (resp. right) $A$-module structure. Moreover, the scalar multiplication of this structure is given by
    \[
        a \cdot x = \eta(a) \cdot x \quad \text{for all} \; a \in A, x \in M.
    \]
\end{proposition}

\begin{proof}
    Assume that $M, N$ are left modules, The right modules case is analogous. 
    
    Let $a, b \in A$ and $x, y \in M$, we have:
    \begin{enumerate}[label=(\roman*)]
        \item $a (x + y) = \eta(a) (x + y) = \eta(a) x + \eta(a) y = a x + a y$,
        \item $(a + b) x = \eta(a + b) x = (\eta(a) + \eta(b)) x = \eta(a) x + \eta(b) x = a x + b x $,
        \item $(a b) x = \eta(a b) x = (\eta(a) \eta(b)) x = \eta(a) (\eta(b) x) = a (b x)$,
        \item $1_{A} x = \eta(1_A) x = 1_B x = x$.
    \end{enumerate}
    Hence, this induces a left $A$-module structure on $M$.
\end{proof}

\begin{remark}
    Given $A, B$ be two nonzero rings. A ring morphism $A \to B$ induces a left (resp. right) $A$-module structure on $B$ and on any of its left (resp. right) ideal.
\end{remark}
