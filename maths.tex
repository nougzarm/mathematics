\subsection{Introduction to groups}
\subsubsection{Magmas and properties}

\begin{definition} \label{def-magma}
    A \emph{magma} is a pair $(M, \ast)$ with $M$ a set and 
    $$
        \ast \colon M \times M \to M
    $$
    an internal composition law.
\end{definition}

\begin{definition} \label{def-magma-associative}
    A magma $(M, \ast)$ is said to be \emph{associative} if $\forall a, b, c \in M, \; (a \ast b) \ast c = a \ast (b \ast c)$.
\end{definition}

\begin{definition} \label{def-magma-commutative}
    A magma $(M, \ast)$ is said to be \emph{commutative} if $\forall a, b \in M, \; a \ast b = b \ast a$.
\end{definition}

\begin{definition} \label{def-magma-id-element}
    An \emph{identity element} of a given magma $(M, \ast)$ is an element $e \in M$, such that $\forall a \in M, \; e \ast a = a \ast e = a$.
\end{definition}

\begin{theorem} \label{thm-magma-id-element-uniqueness}
    If a magma $(M, \ast)$ has an identity element, then it is unique.
\end{theorem}

\begin{proof}
    If $e, e^{\prime} \in M$ are identity elements for $\ast$, then $e = e \ast e^{\prime} = e^{\prime}$.
\end{proof}

\begin{definition} \label{def-magma-invertible-element}
    Given a magma $(M, \ast)$ with an identity element $e \in M$. An element $a \in M$ is said to be \emph{invertible} if there exists $b \in M$ such that $a \ast b = b \ast a = e$. $b$ is then called an inverse of $a$.
\end{definition}

\subsubsection{Products of magmas}

\begin{definition} \label{def-magma-product}
    Given two magmas $(M, \ast)$ and $(N, \bullet)$, we define a canonical structure of magma on the set $M \times N$, defined by the following internal composition law:
    $$
        \big((a, b), (a^{\prime}, b^{\prime}) \big) \mapsto (a \ast a^{\prime}, b \bullet b^{\prime})
    $$
\end{definition}

\begin{proposition} \label{prop-magma-product-props}
    Let $M$ and $N$ be two magmas.
    \begin{enumerate}[label=(\roman*)]
        \item If $M$ and $N$ are associative (resp. commutative), then $M \times N$ is associative (resp. commutative),
        \item If $M$ and $N$ have identity elements, then $M \times N$ has an identity element.
    \end{enumerate}
\end{proposition}

\begin{proof}
    Admitted.
\end{proof}

\begin{proposition} \label{prop-magma-product-reversibility}
    Let $M$ and $N$ be two magmas with identity elements. If $a \in M$ and $b \in N$ are invertible, then $(a, b) \in M \times N$ is invertible.
\end{proposition}

\begin{proof}
    Admitted.
\end{proof}

\begin{remark}
    The previous construction can be extended to any product of magmas. The propositions \eqref{prop-magma-product-props} and \eqref{prop-magma-product-reversibility} then remain true.
\end{remark}

\subsubsection{Morphism of magmas}

\begin{definition} \label{def-magma-morphism}
    A \emph{morphism of magmas} between $(M, \ast)$ and $(N, \bullet)$ is a map $f \colon M \to N$ such that
    $$
        \forall a, b \in M, \; f(a \ast b) = f(a) \bullet f(b)
    $$
\end{definition}

\begin{proposition} \label{prop-magma-morphism-composition}
    A composition of morphisms of magmas is a morphism of magmas.
\end{proposition}

\begin{proof}
    Admitted.
\end{proof}

\subsubsection{Monoids and properties}

\begin{definition} \label{def-monoid}
    A \emph{monoid} is an associative magma with an identity element.
\end{definition}

\begin{theorem} \label{thm-monoid-inverse-uniqueness}
    If $(M, \ast)$ is a monoid, then every invertible element $a \in M$ has a unique inverse, denoted by $a^{-1}$.
\end{theorem}

\begin{proof}
    Let $b, b^{\prime} \in M$ be two inverses of $a$. Then we have
    $$
        b = b \ast (a \ast b^{\prime}) = (b \ast a) \ast b^{\prime} = b^{\prime}.
    $$
\end{proof}

\begin{notation} \label{not-monoid-exposant}
    If $(M, \ast)$ is a monoid, then for all $a \in M$, we denote:
    $$
        a^0 = e \quad \text{and} \quad a^n = \underbrace{a \ast \dots \ast a}_{n \text{ times}} \quad (\text{for } n \geq 1).
    $$
\end{notation}

\begin{proposition}
    Given a monoid $(M, \ast)$, and $a, x, y \in M$,
    \begin{enumerate}[label=(\roman*)]
        \item If $a \ast x = a \ast y$ and $a$ is invertible, then $x = y$,
        \item If $x$ is invertible, then $x^{-1}$ is invertible and $(x^{-1})^{-1} = x$.
        \item If $x$ and $y$ are invertible, then $x \ast y$ is also invertible and $(x \ast y)^{-1} = y^{-1} \ast x^{-1}$,
        \item If $x$ is invertible, then for all $n \in \mathbb{N}, \; x^n$ is also invertible and $(x^n)^{-1} = (x^{-1})^n$. Hence we can denote $x^{-n} = (x^n)^{-1}$,
    \end{enumerate}
\end{proposition}

\begin{proof}
    \begin{enumerate}[label=(\roman*)]
        \item $x = (a^{-1} \ast a) \ast x = a^{-1} \ast (a \ast x) = a^{-1} \ast (a \ast y) = (a^{-1} \ast a) \ast y = y$.
        \item Direct consequence of the invertible element definition.
        \item $(x \ast y) \ast (y^{-1} \ast x^{-1}) = x \ast y \ast y^{-1} \ast x^{-1} = e$, and similarly for the other way.
        \item It suffices to combine the two preceding properties in a proof by induction.
    \end{enumerate}
\end{proof}

\begin{proposition} \label{prop-monoid-product}
    Any product of monoids is a monoid.
\end{proposition}

\begin{proof}
    It is induced by \eqref{prop-magma-product-props}.
\end{proof}

\begin{definition} \label{def-monoid-morphism}
    A \emph{morphism of monoids} is a morphism between underlying magmas, such that it preserves the identity element.
\end{definition}

\subsubsection{Groups}

\begin{definition} \label{def-group}
    A \emph{group} is a monoid such that every element is invertible.
\end{definition}

\begin{definition} \label{def-group-order}
    The order of a group $G$, denoted by $\lvert G \rvert$ or $\# G$, is the cardinality of its underlying set. 
\end{definition}

\begin{definition} \label{def-group-commutative}
    A group is \emph{commutative} or \emph{abelian} if the underlying magma is commutative.
\end{definition}

\begin{proposition} \label{prop-group-product}
    Any product of groups is a group.
\end{proposition}

\begin{proof}
    It is induced by \eqref{prop-monoid-product} and \eqref{prop-magma-product-reversibility}.
\end{proof}

\subsection{Subgroups, morphism of groups}
\subsubsection{Subgroups}

\begin{definition} \label{def-subgroup}
    Let $(G, \ast)$ be a group. A subset $H \subseteq G$ is said to be a \emph{subgroup} of $G$ if:
    \begin{enumerate}[label=(\roman*)]
        \item The identity element $e \in H$,
        \item $H$ is closed under the composition law, i.e., $\forall x, y \in H, \; x \ast y \in H$,
        \item $H$ is closed under taking inverses, i.e., $\forall x \in H, \; x^{-1} \in H$.
    \end{enumerate}
\end{definition}

\begin{proposition} \label{prop-subgroup-criterion}
    Let $(G, \ast)$ be a group. A non-empty subset $H \subseteq G$ is a subgroup of $G$ if and only if:
    $$
        \forall x, y \in H, \quad x \ast y^{-1} \in H.
    $$
\end{proposition}

\begin{proof}
    Suppose $H$ is a subgroup of $G$. Let $x, y \in H$. By condition (iii) of Definition~\ref{def-subgroup}, $y^{-1} \in H$. Then, by condition (ii), $x \ast y^{-1} \in H$.\\
    Conversely, suppose $H \neq \emptyset$ and satisfies the given criterion. Since $H$ is non-empty, there exists some $x \in H$. Applying the criterion to $x$ and $x$ yields $x \ast x^{-1} = e \in H$, which satisfies condition (i). Now, for any $y \in H$, applying the criterion to $e$ and $y$ gives $e \ast y^{-1} = y^{-1} \in H$, satisfying condition (iii). Finally, for any $x, y \in H$, we know $y^{-1} \in H$, so applying the criterion to $x$ and $y^{-1}$ gives $x \ast (y^{-1})^{-1} = x \ast y \in H$, satisfying condition (ii).
\end{proof}

\begin{proposition} \label{prop-group-subgroups-intersection}
    Given a group $G$, any intersection of subgroups of $G$ is a subgroup of $G$.
\end{proposition}

\begin{proof}
    Admitted.
\end{proof}

\subsubsection{Morphism of groups}

\begin{definition} \label{def-group-morphism}
    A \emph{morphism of groups} is a morphism of underlying magmas.
\end{definition}

\begin{definition} \label{def-group-morphism-domain}
    Let $f \colon G \to G^{\prime}$ be a morphism of groups. We say that $G$ is the \emph{domain} of $f$, and $G^{\prime}$ is the \emph{codomain} of $f$.
\end{definition}

\begin{proposition} \label{prop-group-morphism-reversibility}
    Let $G, G^{\prime}$ be two groups and $f \colon G \to G^{\prime}$ be a morphism of groups. Then $f(e_{G}) = e_{G^{\prime}}$ and for all $x \in G, \; f(x^{-1}) = f(x)^{-1}$.
\end{proposition}

\begin{proof}
    Admitted.
\end{proof}

\begin{proposition} \label{prop-group-morphism-composition}
    A composition of morphisms of groups is a morphism of groups.
\end{proposition}

\begin{proof}
    Induced by \eqref{prop-magma-morphism-composition}.
\end{proof}

\begin{theorem} \label{thm-group-morphism-image}
    Let $G, G^{\prime}$ be two groups and $f \colon G \to G^{\prime}$ be a morphism of groups.
    \begin{enumerate}[label=(\roman*)]
        \item If $H$ is a subgroup of $G$, then $f(H)$ is a subgroup of $G^{\prime}$,
        \item If $H^{\prime}$ is a subgroup of $G^{\prime}$, then $f^{-1}(H^{\prime})$ is a subgroup of $G$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    Admitted.
\end{proof}

\begin{definition} \label{def-group-morphism-kernel}
    The \emph{kernel} of a morphism of groups $f \colon G \to G^{\prime}$ is defined by
    $$
        \text{Ker}(f) = \lbrace a \in G \mid f(a) = e_{G^{\prime}} \rbrace
    $$
\end{definition}

\begin{proposition} \label{prop-group-morphism-kernel-subgroup}
    The kernel of a morphism of groups $f \colon G \to G^{\prime}$ is a subgroup of $G$.
\end{proposition}

\begin{proof}
    Induced by \eqref{thm-group-morphism-image}-(ii).
\end{proof}

\begin{proposition} \label{prop-group-morphism-inj-criterion}
    A morphism of groups $f \colon G \to G^{\prime}$ is injective if and only if its kernel is the trivial subgroup $\lbrace e_{G} \rbrace$.
\end{proposition}

\begin{proof}
    The if condition: Let $x, x^{\prime} \in G$ be such that $f(x) = f(x^{\prime})$, which is equivalent to $f(x) f(x^{\prime})^{-1} = e_{G^{\prime}}$. However, $f(x) f(x^{\prime})^{-1} = f(x) f((x^{\prime})^{-1}) = f(x (x^{\prime})^{-1})$. Which implies, by hypothesis $x (x^{\prime})^{-1} = e_{G}$. So $x = x^{\prime}$. The reverse is obvious.
\end{proof}

\begin{definition} \label{def-group-morphism-image}
    The \emph{image} of a morphism of groups $f \colon G \to G^{\prime}$ is defined by
    $$
        \text{Im}(f) = \lbrace b \in G^{\prime} \mid \exists a \in G, \; f(a) = b \rbrace
    $$
\end{definition}

\begin{proposition} \label{prop-group-morphism-image-subgroup}
    The image of a morphism of groups $f \colon G \to G^{\prime}$ is a subgroup of $G^{\prime}$.
\end{proposition}

\begin{proof}
    Induced by \eqref{thm-group-morphism-image}-(i).
\end{proof}

\subsubsection{Isomorphism of groups}

\begin{definition} \label{def-group-isomorphism}
    An \emph{isomorphism of groups} is a bijective morphism of groups.
\end{definition}

\begin{notation}
    If $f \colon G \to G^{\prime}$ is an isomorphism, then we say that $G$ is isomorphic to $G^{\prime}$. If $G = G^{\prime}$, we say that $f$ is an automorphism.
\end{notation}

\begin{theorem} \label{thm-group-isomorphism-composition}
    The composition of two isomorphisms of groups is an isomorphism of groups.
\end{theorem}

\begin{proof}
    Obvious.
\end{proof}

\begin{theorem} \label{thm-group-isomorphism-inverse}
    If $f$ is an isomorphism of groups, then the inverse $f^{-1}$ is a morphism of groups. Moreover, it is an isomorphism of groups.
\end{theorem}

\begin{proof}
    Let $f \colon G \to G^{\prime}$ be an isomorphism of groups and $y, y^{\prime} \in G^{\prime}$. 
    Then
    $$
    \begin{aligned}
        f^{-1}(y y^{\prime}) 
            &= f^{-1}\Big( f\big( f^{-1}(y) \big) f\big( f^{-1}(y^{\prime}) \big) \Big) \\
            &= f^{-1}\Big( f\big( f^{-1}(y) f^{-1}(y^{\prime}) \big) \Big) \\
            &= f^{-1}(y) f^{-1}(y^{\prime})
    \end{aligned}
    $$
\end{proof}

\begin{proposition}
    The relation "is isomorphic to" is an equivalence relation on the class of groups. Its equivalence classes are called isomorphism classes.
\end{proposition}

\begin{proof}
    Induced by \eqref{thm-group-isomorphism-inverse} and \eqref{thm-group-isomorphism-composition}.
\end{proof}

\subsection{Normal subgroup and quotient}

\begin{definition} \label{def-group-normal-subgroup}
    Let $G$ be a group. A subgroup $H$ of $G$ is said to be a \emph{normal} subgroup if:
    $$
        \forall h \in H, \forall g \in G, \; g h g^{-1} \in H
    $$
\end{definition}

\begin{remark}
    If $G$ is an abelian group, then any subgroup of $G$ is a normal subgroup.
\end{remark}

\begin{definition} \label{def-group-canoncial-surjection}
    Let $G$ be a group and $H$ a subgroup of $G$. For $g \in G$, we denote the subset $gH = \lbrace gh, \; h \in H \rbrace$, called \emph{left coset} of $g$ modulo $H$. The left cosets constitute a partition of $G$; The set of left cosets modulo $H$ is called the \emph{left quotient} of $G$ by $H$, denoted by $G / H$; the map
    $$
        \begin{aligned}
            \pi \colon G &\longrightarrow G / H \\
            g &\longmapsto gH
        \end{aligned}
    $$
    is called \emph{canonical surjection}.
\end{definition}

\begin{definition} \label{def-group-subgroup-index}
    Let $G$ be a group and $H$ a subgroup of $G$. The \emph{index} of $H$ in $G$, denoted by $[G : H]$, is the cardinality of the quotient $G / H$.
\end{definition}

\begin{proposition}
    Let $G$ be a group and $H$ a subgroup of $G$. Then we have
    $$
        \lvert G \rvert = [G : H] \lvert H \rvert
    $$
\end{proposition}

\begin{proof}
    Let $X$ be a subset of $G$ such that the restriction of the canonical surjection $\pi \colon G \to G / H$ to $X$ is bijective. Then we have $[G : H] = \text{card}(X)$ ($X$ is a system of representatives of left cosets of $G$ by $H$).
    Let the map
    $$
        \begin{aligned}
            f \colon X \times H &\longrightarrow G \\
            (x, h) &\longmapsto xh
        \end{aligned}
    $$
    which is a bijection. Hence, $\lvert G \rvert = \text{card} (X \times H) = [G : H] \lvert H \rvert$.
\end{proof}

\begin{remark} \label{rem-group-subgroup-order-divisibility}
    If the order of $G$ is finite, then the order of $H$ divides the order of $G$.
\end{remark}

\begin{proposition} \label{prop-group-quotient}
    Let $G$ be a group and $H$ a subgroup of $G$. The following conditions are equivalent:
    \begin{enumerate}[label=(\roman*)]
        \item $H$ is a normal subgroup,
        \item There exists a group structure on the quotient $G / H$, such that the canonical surjection $G \to G / H$ is a morphism of groups,
        \item There exists a group $G^{\prime}$ and a morphism of groups $f \colon G \to G^{\prime}$ such that $H$ is the kernel of $f$.
    \end{enumerate}
\end{proposition}

\begin{proof}
    Admitted.
\end{proof}
