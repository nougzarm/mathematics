\subsubsection{Magmas and properties}

\begin{definition} \label{def-magma}
    A \emph{magma} is a pair $(M, \ast)$ with $M$ a set and 
    $$
        \ast \colon M \times M \to M
    $$
    an internal composition law.
\end{definition}

\begin{definition} \label{def-magma-associative}
    A magma $(M, \ast)$ is said to be \emph{associative} if $\forall a, b, c \in M, \; (a \ast b) \ast c = a \ast (b \ast c)$.
\end{definition}

\begin{definition} \label{def-magma-commutative}
    A magma $(M, \ast)$ is said to be \emph{commutative} if $\forall a, b \in M, \; a \ast b = b \ast a$.
\end{definition}

\begin{definition} \label{def-magma-id-element}
    An \emph{identity element} of a given magma $(M, \ast)$ is an element $e \in M$, such that $\forall a \in M, \; e \ast a = a \ast e = a$.
\end{definition}

\begin{theorem} \label{thm-magma-id-element-uniqueness}
    If a magma $(M, \ast)$ has an identity element, then it is unique.
\end{theorem}

\begin{proof}
    If $e, e^{\prime} \in M$ are identity elements for $\ast$, then $e = e \ast e^{\prime} = e^{\prime}$.
\end{proof}

\begin{definition} \label{def-magma-invertible-element}
    Given a magma $(M, \ast)$ with an identity element $e \in M$. An element $a \in M$ is said to be \emph{invertible} if there exists $b \in M$ such that $a \ast b = b \ast a = e$. $b$ is then called an inverse of $a$.
\end{definition}

\subsubsection{Submagmas}
\begin{definition} \label{def-magma-submagma}
    Let $(M, \ast)$ be a magma. A subset $N \subset M$ is a \emph{submagma} of $(M, \ast)$ if the operation $\ast$ is stable under $N$:
    $$
        \forall a, b \in N, \; a \ast b \in N.
    $$
\end{definition}

\begin{remark} \label{rem-magma-canoncial-submagma}
    The submagma $N$ induces the canonical magma $(N, \ast_{\vert N \times N})$.
\end{remark}

\begin{remark} \label{rem-magma-submagmas-interesection}
    Any intersection of submagmas of a magma $(M, \ast)$ is a submagma of $(M, \ast)$.
\end{remark}

\subsubsection{Products of magmas}

\begin{definition} \label{def-magma-product}
    Given two magmas $(M, \ast)$ and $(N, \cdot)$, we define a canonical structure of magma on the set $M \times N$, defined by the following internal composition law:
    $$
        \big((a, b), (a^{\prime}, b^{\prime}) \big) \mapsto (a \ast a^{\prime}, b \cdot b^{\prime})
    $$
\end{definition}

\begin{proposition} \label{prop-magma-product-props}
    Let $M$ and $N$ be two magmas.
    \begin{enumerate}[label=(\roman*)]
        \item If $M$ and $N$ are associative (resp. commutative), then $M \times N$ is associative (resp. commutative),
        \item If $M$ and $N$ have identity elements, then $M \times N$ has an identity element.
    \end{enumerate}
\end{proposition}

\begin{proof}
    Omitted.
\end{proof}

\begin{proposition} \label{prop-magma-product-reversibility}
    Let $M$ and $N$ be two magmas with identity elements. If $a \in M$ and $b \in N$ are invertible, then $(a, b) \in M \times N$ is invertible.
\end{proposition}

\begin{proof}
    Omitted.
\end{proof}

\begin{remark}
    The previous construction can be extended to any product of magmas. The propositions \eqref{prop-magma-product-props} and \eqref{prop-magma-product-reversibility} then remain true.
\end{remark}

\subsubsection{Magma morphisms}

\begin{definition} \label{def-magma-morphism}
    A \emph{magma morphism} from $(M, \ast)$ to $(N, \cdot)$ is a map $f \colon M \to N$ such that:
    $$
        \forall a, b \in M, \; f(a \ast b) = f(a) \cdot f(b)
    $$
\end{definition}

\begin{proposition} \label{prop-magma-morphism-composition}
    A composition of magma morphisms is a magma morphism.
\end{proposition}

\begin{proof}
    Omitted.
\end{proof}

\begin{theorem} \label{thm-magma-morphism-image}
    Let $f \colon (M, \ast) \to (M^{\prime}, \cdot)$ be a magma morphism, then:
    \begin{enumerate}[label=(\roman*)]
        \item If $N$ is a submagma of $M$, then $f(N)$ is a submagma of $M^{\prime}$,
        \item If $N^{\prime}$ is a submagma of $M^{\prime}$, then $f^{-1}(N^{\prime})$ is a submagma of $M$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    Firstly, for all $a, b \in N$, we have: $f(a) \cdot f(b) = f(a \ast b) \in f(N)$. Secondly, for all $a, b \in f^{-1}(N^{\prime})$, we have: $f(a \ast b) = f(a) \cdot f(b) \in N^{\prime}$.
\end{proof}

\begin{definition} \label{def-magma-isomorphism}
    A \emph{magma isomorphism} is a bijective magma morphism.
\end{definition}

\begin{theorem} \label{thm-magma-isomorphism-inverse}
    If $f$ is a magma isomorphism, then the inverse $f^{-1}$ is a magma morphism. Moreover, it is a magma isomorphism.
\end{theorem}

\begin{proof}
    Let $f \colon (M, \ast) \to (M^{\prime}, \cdot)$ be a magma isomorphism and $a, a^{\prime} \in M^{\prime}$. 
    Then
    $$
    \begin{aligned}
        f^{-1}(a \cdot a^{\prime}) 
            &= f^{-1}\Big( f\big( f^{-1}(a) \big) \cdot f\big( f^{-1}(a^{\prime}) \big) \Big) \\
            &= f^{-1}\Big( f\big( f^{-1}(a) \ast f^{-1}(a^{\prime}) \big) \Big) \\
            &= (f^{-1} \circ f) \Big( f^{-1}(a) \ast f^{-1}(a^{\prime}) \Big) \\
            &= f^{-1}(a) \ast f^{-1}(a^{\prime})
    \end{aligned}
    $$
\end{proof}

\subsubsection{Monoids and properties}

\begin{definition} \label{def-monoid}
    A \emph{monoid} is an associative magma with an identity element.
\end{definition}

\begin{theorem} \label{thm-monoid-inverse-uniqueness}
    If $(M, \ast)$ is a monoid, then every invertible element $a \in M$ has a unique inverse, denoted by $a^{-1}$.
\end{theorem}

\begin{proof}
    Let $b, b^{\prime} \in M$ be two inverses of $a$. Then we have
    $$
        b = b \ast (a \ast b^{\prime}) = (b \ast a) \ast b^{\prime} = b^{\prime}.
    $$
\end{proof}

\begin{notation} \label{not-monoid-exposant}
    If $(M, \ast)$ is a monoid, then for all $a \in M$, we denote:
    $$
        a^0 = e \quad \text{and} \quad a^n = \underbrace{a \ast \dots \ast a}_{n \text{ times}} \quad (\text{for } n \geq 1).
    $$
\end{notation}

\begin{proposition}
    Given a monoid $(M, \ast)$, and $a, x, y \in M$,
    \begin{enumerate}[label=(\roman*)]
        \item If $a \ast x = a \ast y$ and $a$ is invertible, then $x = y$,
        \item If $x$ is invertible, then $x^{-1}$ is invertible and $(x^{-1})^{-1} = x$.
        \item If $x$ and $y$ are invertible, then $x \ast y$ is also invertible and $(x \ast y)^{-1} = y^{-1} \ast x^{-1}$,
        \item If $x$ is invertible, then for all $n \in \mathbb{N}, \; x^n$ is also invertible and $(x^n)^{-1} = (x^{-1})^n$. Hence we can denote $x^{-n} = (x^n)^{-1}$,
    \end{enumerate}
\end{proposition}

\begin{proof}
    \begin{enumerate}[label=(\roman*)]
        \item $x = (a^{-1} \ast a) \ast x = a^{-1} \ast (a \ast x) = a^{-1} \ast (a \ast y) = (a^{-1} \ast a) \ast y = y$.
        \item Direct consequence of the invertible element definition.
        \item $(x \ast y) \ast (y^{-1} \ast x^{-1}) = x \ast y \ast y^{-1} \ast x^{-1} = e$, and similarly for the other way.
        \item It suffices to combine the two preceding properties in a proof by induction.
    \end{enumerate}
\end{proof}

\subsubsection{Submonoid, product and morphism}

\begin{definition} \label{def-monoid-submonoid}
    Let $(M, \ast)$ be a monoid. A subset $N \subset M$ is a \emph{submonoid} of $(M, \ast)$ if it is its submagma and containing the identity element.
\end{definition}

\begin{remark} \label{rem-monoid-canoncial-submonoid}
    The submonoid $N$ induces the canonical monoid $(N, \ast_{\vert N \times N})$.
\end{remark}

\begin{remark} \label{rem-monoid-submonoids-interesection}
    Any intersection of submonoids of a monoid $(M, \ast)$ is a submonoid of $(M, \ast)$.
\end{remark}

\begin{proof}
    Obvious by \eqref{rem-magma-submagmas-interesection} and the definition \eqref{def-monoid-submonoid}
\end{proof}

\begin{proposition} \label{prop-monoid-product}
    Any product of monoids is a monoid.
\end{proposition}

\begin{proof}
    It is induced by \eqref{prop-magma-product-props}.
\end{proof}

\begin{definition} \label{def-monoid-morphism}
    A \emph{monoid morphism} is a morphism between underlying magmas, such that it preserves the identity element.
\end{definition}

\begin{proposition} \label{prop-monoid-morphism-composition}
    A composition of monoid morphisms is a monoid morphism.
\end{proposition}

\begin{proof}
    By \eqref{prop-magma-morphism-composition} and the fact that the invariance is stable under composition.
\end{proof}

\begin{theorem} \label{thm-monoid-morphism-image}
    Let $f \colon M \to M^{\prime}$ be a monoid morphism, then:
    \begin{enumerate}[label=(\roman*)]
        \item If $N$ is a submonoid of $M$, then $f(N)$ is a submonoid of $M^{\prime}$,
        \item If $N^{\prime}$ is a submonoid of $M^{\prime}$, then $f^{-1}(N^{\prime})$ is a submonoid of $M$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    Induced by \eqref{thm-magma-morphism-image} and the fact that a monoid morphism preserves the identity.
\end{proof}

\begin{proposition} \label{prop-monoid-invertible-image}
    Let $f \colon (M, \ast) \to (N, \cdot)$ be a monoid morphism. If $a \in M$ is invertible, then $f(a) \in N$ is also invertible.
\end{proposition}

\begin{proof}
    $$
        f(a) \cdot f(a^{-1}) = f(a \ast a^{-1}) = f(e_{M}) = e_{N}.
    $$
    So, $f(a)$ is invertible, and its inverse is $f(a^{-1})$.
\end{proof}

\begin{definition} \label{def-monoid-isomorphism}
    A \emph{monoid isomorphism} is a bijective monoid morphism.
\end{definition}

\begin{theorem} \label{thm-monoid-isomorphism-inverse}
    If $f$ is a monoid isomorphism, then the inverse $f^{-1}$ is a monoid morphism. Moreover, it is a monoid isomorphism.
\end{theorem}

\begin{proof}
    By \eqref{thm-magma-isomorphism-inverse} and the fact that $f^{-1}$ preserves also identity.
\end{proof}

\subsubsection{Groups}

\begin{definition} \label{def-group}
    A \emph{group} is a monoid such that every element is invertible.
\end{definition}

\begin{definition} \label{def-group-order}
    The order of a group $G$, denoted by $\lvert G \rvert$ or $\# G$, is the cardinality of its underlying set. 
\end{definition}

\begin{definition} \label{def-group-commutative}
    A group is \emph{commutative} or \emph{abelian} if the underlying magma is commutative.
\end{definition}

\begin{proposition} \label{prop-group-product}
    Any product of groups is a group.
\end{proposition}

\begin{proof}
    It is induced by \eqref{prop-monoid-product} and \eqref{prop-magma-product-reversibility}.
\end{proof}

\subsection{Subgroups, group morphisms}
\subsubsection{Subgroups}

\begin{definition} \label{def-subgroup}
    Let $(G, \ast)$ be a group. A subset $H \subseteq G$ is said to be a \emph{subgroup} of $G$ if it is a submonoid and is stable by inversion: $\forall x \in H, \; x^{-1} \in H$.
\end{definition}

\begin{remark} \label{rem-group-canoncial-subgroup}
    The subgroup $H$ induces the canonical group $(H, \ast_{\vert H \times H})$.
\end{remark}

\begin{proposition} \label{prop-subgroup-criterion}
    Let $(G, \ast)$ be a group. A non-empty subset $H \subseteq G$ is a subgroup of $G$ if and only if:
    $$
        \forall x, y \in H, \quad x \ast y^{-1} \in H.
    $$
\end{proposition}

\begin{proof}
    The direct way is a consequence of the definition of subgroups.\\
    Conversely, suppose $H \neq \emptyset$ and satisfies the given criterion. Since $H$ is non-empty, there exists some $x \in H$. Applying the criterion to $x$ and $x$ yields $x \ast x^{-1} = e \in H$. Now, for any $y \in H$, applying the criterion to $e$ and $y$ gives $e \ast y^{-1} = y^{-1} \in H$. Finally, for any $x, y \in H$, we know $y^{-1} \in H$, so applying the criterion to $x$ and $y^{-1}$ gives $x \ast (y^{-1})^{-1} = x \ast y \in H$.
\end{proof}

\begin{proposition} \label{prop-group-subgroups-intersection}
    Given a group $G$, any intersection of subgroups of $G$ is a subgroup of $G$.
\end{proposition}

\begin{proof}
    By \eqref{rem-monoid-submonoids-interesection} and the fact that stability under inversion is preserved after intersection.
\end{proof}

\subsubsection{Group morphisms}

\begin{definition} \label{def-group-morphism}
    A \emph{group morphism} is a morphism of underlying magmas.
\end{definition}

\begin{notation} \label{not-group-operation}
    From now on, the group operation will be denoted multiplicatively. 
    Hence, for any $x, y \in G$, the product $x \ast y$ will be implicitly written as $xy$. 
    When the context requires it (notably for abelian groups), the additive notation $x + y$ will be preferred.
\end{notation}

\begin{definition} \label{def-group-morphism-domain}
    Let $f \colon G \to G^{\prime}$ be a group morphism. We say that $G$ is the \emph{domain} of $f$, and $G^{\prime}$ is the \emph{codomain} of $f$.
\end{definition}

\begin{proposition} \label{prop-group-morphism-reversibility}
    Let $G, G^{\prime}$ be two groups and $f \colon G \to G^{\prime}$ be a group morphism. Then $f(e_{G}) = e_{G^{\prime}}$ and for all $x \in G, \; f(x^{-1}) = f(x)^{-1}$.
\end{proposition}

\begin{proof}
    Omitted.
\end{proof}

\begin{proposition} \label{prop-group-morphism-composition}
    A composition of group morphisms is a group morphism.
\end{proposition}

\begin{proof}
    Induced by \eqref{prop-magma-morphism-composition}.
\end{proof}

\begin{theorem} \label{thm-group-morphism-image}
    Let $f \colon G \to G^{\prime}$ be a group morphism.
    \begin{enumerate}[label=(\roman*)]
        \item If $H$ is a subgroup of $G$, then $f(H)$ is a subgroup of $G^{\prime}$,
        \item If $H^{\prime}$ is a subgroup of $G^{\prime}$, then $f^{-1}(H^{\prime})$ is a subgroup of $G$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    Induced by \eqref{thm-monoid-morphism-image} and \eqref{prop-group-morphism-reversibility}.
\end{proof}

\begin{definition} \label{def-group-morphism-kernel}
    The \emph{kernel} of a group morphism $f \colon G \to G^{\prime}$ is defined by
    $$
        \text{Ker}(f) = \lbrace a \in G \mid f(a) = e_{G^{\prime}} \rbrace
    $$
\end{definition}

\begin{proposition} \label{prop-group-morphism-kernel-subgroup}
    The kernel of a group morphism $f \colon G \to G^{\prime}$ is a subgroup of $G$.
\end{proposition}

\begin{proof}
    Induced by \eqref{thm-group-morphism-image}-(ii).
\end{proof}

\begin{proposition} \label{prop-group-morphism-inj-criterion}
    A morphism of groups $f \colon G \to G^{\prime}$ is injective if and only if its kernel is the trivial subgroup $\lbrace e_{G} \rbrace$.
\end{proposition}

\begin{proof}
    The if condition: Let $x, x^{\prime} \in G$ be such that $f(x) = f(x^{\prime})$, which is equivalent to $f(x) f(x^{\prime})^{-1} = e_{G^{\prime}}$. However, $f(x) f(x^{\prime})^{-1} = f(x) f((x^{\prime})^{-1}) = f(x (x^{\prime})^{-1})$. Which implies, by hypothesis $x (x^{\prime})^{-1} = e_{G}$. So $x = x^{\prime}$. The reverse is obvious.
\end{proof}

\begin{definition} \label{def-group-morphism-image}
    The \emph{image} of a group morphism is the set image of the underlying map.
\end{definition}

\begin{proposition} \label{prop-group-morphism-image-subgroup}
    The image of a group morphism $f \colon G \to G^{\prime}$ is a subgroup of $G^{\prime}$.
\end{proposition}

\begin{proof}
    Induced by \eqref{thm-group-morphism-image}-(i).
\end{proof}

\subsubsection{Isomorphism of groups}

\begin{definition} \label{def-group-isomorphism}
    A \emph{group isomorphism} is a bijective group morphism.
\end{definition}

\begin{notation}
    If $f \colon G \to G^{\prime}$ is an isomorphism, then we say that $G$ is isomorphic to $G^{\prime}$. If $G = G^{\prime}$, we say that $f$ is an automorphism.
\end{notation}

\begin{theorem} \label{thm-group-isomorphism-composition}
    The composition of two group isomorphisms is a group isomorphism.
\end{theorem}

\begin{proof}
    Obvious.
\end{proof}

\begin{theorem} \label{thm-group-isomorphism-inverse}
    If $f$ is a group isomorphism, then the inverse $f^{-1}$ is a group morphism. Moreover, it is a group isomorphism.
\end{theorem}

\begin{proof}
    Induced by \eqref{thm-magma-isomorphism-inverse}.
\end{proof}

\begin{proposition}
    The relation "is isomorphic to" is an equivalence relation on the class of groups. Its equivalence classes are called \emph{isomorphism classes}.
\end{proposition}

\begin{proof}
    Induced by \eqref{thm-group-isomorphism-inverse} and \eqref{thm-group-isomorphism-composition}.
\end{proof}

\subsection{Normal subgroup and quotient}

\begin{definition} \label{def-group-normal-subgroup}
    Let $G$ be a group. A subgroup $H$ of $G$ is said to be a \emph{normal} subgroup if:
    $$
        \forall h \in H, \forall g \in G, \; g h g^{-1} \in H
    $$
\end{definition}

\begin{remark}
    If $G$ is an abelian group, then any subgroup of $G$ is a normal subgroup.
\end{remark}

\begin{definition} \label{def-group-canoncial-surjection}
    Let $G$ be a group and $H$ a subgroup of $G$. For $g \in G$, we denote the subset $gH = \lbrace gh \mid h \in H \rbrace$, called the \emph{left coset} of $g$ modulo $H$. The left cosets constitute a partition of $G$; The set of left cosets modulo $H$ is called the \emph{left quotient} of $G$ by $H$, denoted by $G / H$; the map
    $$
        \begin{aligned}
            \pi \colon G &\longrightarrow G / H \\
            g &\longmapsto gH
        \end{aligned}
    $$
    is called the \emph{canonical surjection}.
\end{definition}

\begin{definition} \label{def-group-subgroup-index}
    Let $G$ be a group and $H$ a subgroup of $G$. The \emph{index} of $H$ in $G$, denoted by $[G : H]$, is the cardinality of the quotient $G / H$.
\end{definition}

\begin{proposition}
    Let $G$ be a group and $H$ a subgroup of $G$. Then we have
    $$
        \lvert G \rvert = [G : H] \lvert H \rvert
    $$
\end{proposition}

\begin{proof}
    Let $X$ be a subset of $G$ such that the restriction of the canonical surjection $\pi \colon G \to G / H$ to $X$ is bijective. Then we have $[G : H] = \text{card}(X)$ ($X$ is a system of representatives of left cosets of $G$ by $H$).
    Let the map
    $$
        \begin{aligned}
            f \colon X \times H &\longrightarrow G \\
            (x, h) &\longmapsto xh
        \end{aligned}
    $$
    which is a bijection. Hence, $\lvert G \rvert = \text{card} (X \times H) = [G : H] \lvert H \rvert$.
\end{proof}

\begin{remark} \label{rem-group-subgroup-order-divisibility}
    If the order of $G$ is finite, then the order of $H$ divides the order of $G$.
\end{remark}

\begin{proposition} \label{prop-group-quotient}
    Let $G$ be a group and $H$ a subgroup of $G$. The following conditions are equivalent:
    \begin{enumerate}[label=(\roman*)]
        \item $H$ is a normal subgroup,
        \item There exists a group structure on the quotient $G / H$, such that the canonical surjection $G \to G / H$ is a morphism of groups,
        \item There exists a group $G^{\prime}$ and a morphism of groups $f \colon G \to G^{\prime}$ such that $H$ is the kernel of $f$.
    \end{enumerate}
\end{proposition}

\begin{proof}
    Omitted.
\end{proof}

\section{Rings}
\subsection{Generalities}
\subsubsection{Rings and subrings}

\begin{definition} \label{def-ring}
    A \emph{ring} is a triplet $(A, +, \cdot)$ such that $(A, +)$ is an abelian group, $(A, \cdot)$ is a monoid, and the operation $\cdot$ is distributive over $+$:
    $$
        \forall a, b, c \in A, \quad a \cdot (b + c) = a \cdot b + a \cdot c \quad \text{and} \quad (b + c) \cdot a = b \cdot a + c \cdot a
    $$
\end{definition}

\begin{notation} \label{not-ring-identities}
    The identity elements are denoted by: $0$ for $(A, +)$ and $1$ for $(A, \cdot)$.
\end{notation}

\begin{notation} \label{not-ring}
    Most often, a ring $(A, +, \cdot)$ is simply denoted by $A$.
\end{notation}

\begin{remark}
    Let $A$ be a ring. Then for all $a \in A$:
    $$
        0 \cdot a = a \cdot 0 = 0 \quad \text{and} \quad (-1) \cdot a = -a = a \cdot (-1)
    $$
\end{remark}

\begin{remark} \label{rem-ring-trivial-ring}
    In a ring $A$, we have $0 = 1$ if, and only if, $A = \lbrace 0 \rbrace$.
\end{remark}

\begin{definition} \label{def-ring-invertible}
    An element of a ring $A$ is \emph{invertible} if it is invertible in $(A, \cdot)$. We denote by $A^{\ast}$ the set of invertible elements of $A$.
\end{definition}

\begin{definition} \label{def-ring-field}
    Let $A$ be a ring. We say that $A$ is a \emph{field} if $A^{\ast} = A \setminus \lbrace 0 \rbrace$.
\end{definition}

\begin{definition} \label{def-ring-subring}
    Let $A$ be a ring. A subset $B \subset A$ is a \emph{subring} of $A$ if it is a subgroup of $(A, +)$ and submonoid of $(A, \cdot)$.
\end{definition}

\begin{remark}
    A subring $B \subset A$ induces the canonical ring $(B, +_{\vert B \times B}, \cdot_{\vert B \times B})$.
\end{remark}

\begin{proposition} \label{prop-ring-generated}
    Let $A$ be a ring. The intersection of any family of subrings of $A$ is also a subring of $A$. In particular, if $X$ is a subset of $A$, the intersection of subrings of $A$ containing $X$ is the smallest subring of $A$ containing $X$, called the \emph{subring} of $A$ \emph{generated} by $X$.
\end{proposition}

\subsubsection{Ring morphisms}

\begin{definition} \label{def-ring-morphism}
    Let $A$ and $B$ be two rings. A map $f \colon A \to B$ is a \emph{ring morphism} if it is a group morphism from $(A, +)$ to $(B, +)$ and a monoid morphism from $(A, \cdot)$ to $(B, \cdot)$.
\end{definition}

\begin{proposition} \label{prop-ring-morphism-composition}
    A composition of ring morphisms is a ring morphism.
\end{proposition}

\begin{proof}
    By \eqref{prop-group-morphism-composition} and \eqref{prop-monoid-morphism-composition}.
\end{proof}

\begin{definition} \label{def-ring-isomorphism}
    An \emph{isomorphism of rings} is a bijective ring morphism.
\end{definition}

\begin{theorem} \label{thm-ring-isomorphism-inverse}
    If $f$ is a ring isomorphism, then the inverse $f^{-1}$ is a ring morphism. Moreover, it is a ring isomorphism.
\end{theorem}

\begin{proof}
    Induced by \eqref{thm-group-isomorphism-inverse} and \eqref{thm-monoid-isomorphism-inverse}.
\end{proof}

\begin{proposition} \label{prop-ring-invertible-image}
    Let $f \colon A \to B$ be a ring morphism. Then $f(A^{\ast}) \subset B^{\ast}$ and $f$ induces a group morphism $(A^{\ast}, \cdot) \to (B^{\ast}, \cdot)$.
\end{proposition}

\begin{proof}
    Induced by \eqref{prop-monoid-invertible-image}.
\end{proof}

\begin{proposition} \label{prop-field-morphism-injectivity}
    Any morphism of fields is injective.
\end{proposition}

\begin{proof}
    Let $f \colon F \to G$ be a morphism between fields. We have $f(F \setminus \lbrace 0 \rbrace) \subset G \setminus \lbrace 0 \rbrace$ by \eqref{prop-ring-invertible-image}, so $f^{-1}(\lbrace 0 \rbrace) \subset \lbrace 0 \rbrace$ which induces that $f$ is injective by \eqref{prop-group-morphism-inj-criterion}.
\end{proof}

\subsubsection{Kernel and Image}

\begin{definition} \label{def-ring-morphism-kernel}
    The \emph{kernel} of a ring morphism is its kernel as a additive group morphism.
\end{definition}

\begin{proposition} \label{prop-ring-morphism-inj-criterion}
    A morphism of rings $f$ is injective if, and only if, $\text{Ker}(f) = \lbrace 0 \rbrace$.
\end{proposition}

\begin{proof}
    Direct consequence of \eqref{prop-ring-morphism-inj-criterion}.
\end{proof}

\begin{definition} \label{def-ring-morphism-image}
    The \emph{image} of a ring morphism is the set image of the underlying map.
\end{definition}

\begin{proposition} \label{prop-ring-morphism-image-subring}
    The image of a ring morphism $f \colon A \to B$ is a subring of $B$.
\end{proposition}

\begin{proof}
    Induced by \eqref{prop-group-morphism-image-subgroup} and \eqref{thm-monoid-morphism-image}-(i).
\end{proof}

\subsection{Ideals and quotients}
\subsubsection{Ideals}

\begin{definition} \label{def-ring-ideal}
    Let $A$ be a ring. A subset $I \subset A$ is an \emph{left ideal} (resp. \emph{right ideal}) if it is a subgroup of $(A, +)$ and stable under external multiplication:
    $$
        \forall a \in A, \forall b \in I, \; a \cdot b \in I \; (\text{resp. } b \cdot a \in I).
    $$
\end{definition}

\begin{definition} \label{def-ring-ideal-two-sided}
    A \emph{two-sided ideal} (or simply \emph{ideal}) is a subset $I \subset A$ that is both a left ideal and a right ideal.
\end{definition}

\begin{remark}
    In a commutative ring, being a left ideal is equivalent to being a right ideal.
\end{remark}

\begin{proposition} \label{prop-ring-morphism-kernel-ideal}
    The kernel of a ring morphism $f \colon A \to B$ is an ideal of $A$.
\end{proposition}

\begin{proof}
    By \eqref{prop-group-morphism-kernel-subgroup} we have that it is a subgroup of $(A, +)$. Moreover, for all $a \in A$ and $b \in \text{Ker}(f)$, we have $f(ab) = f(a) f(b) = 0$ (resp. $f(ba) = 0$), hence $ab$ (resp. $ba$) is in $\text{Ker}(f)$.
\end{proof}

\subsubsection{Quotients}

\begin{proposition}
    Let $A$ be a ring and a additive subgroup $I \subset A$. The following assertions are equivalent:
    \begin{enumerate}[label=(\roman*)]
        \item $I$ is an ideal of $A$,
        \item The group quotient $A / I$ has a ring structure, such that the canonical surjection $\pi \colon A \longrightarrow A / I$ is a ring morphism.
    \end{enumerate}
\end{proposition}

\begin{proof}
    If there exists a ring structure on $A / I$ such that the canonical surjection is a ring morphism, then its kernel, which is $I$, is an ideal of $A$ by \eqref{prop-ring-morphism-kernel-ideal}. 
    
    Conversely, if $I$ is an ideal of $A$: We define the operation $\cdot$ on $A / I$ by:
    $$
    \forall a, b \in A, \; \pi(a) \cdot \pi(b) = \pi(ab)
    $$
    To show that this definition is correct, it requires to show that $\pi(ab) = \pi(a^{\prime} b^{\prime})$ for all $(a, b), (a^{\prime}, b^{\prime}) \in A^2$ such that $\pi(a) = \pi(a^{\prime})$ and $\pi(b) = \pi(b^{\prime})$. However, $\pi(x) = \pi(y)$ if, and only if, $x - y \in I$. Hence, it suffices to show that $ab - a^{\prime} b^{\prime} \in I$. To see it: $ab - a^{\prime} b^{\prime} = a(b - b^{\prime}) + (a - a^{\prime})b^{\prime} \in I$. To finish, the surjectivity of $\pi$ implies that this operation is well-defined on all $A / I$.
\end{proof}
